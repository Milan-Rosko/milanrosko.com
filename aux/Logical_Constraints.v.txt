(*************************************************************************)
(*                                                                       *)
(*  Proofcase / Rule 30 Finite Obstruction — Base Dynamics               *)
(*                                                                       *)
(*  Core Rule-30 semantics on bi-infinite rows and discrete spacetime.   *)
(*  We define the local update rule, global step, iteration, interval    *)
(*  agreement, spatial shifts, and prove locality and shift transport.   *)
(*                                                                       *)
(*************************************************************************)

Section Base_Definitions.

(*
  Binary state of one Rule 30 cell.
*)

Definition cell := bool.

(*
  Bi-infinite row indexed by Z.
*)

Definition row := Z -> cell.

(*
  Discrete spacetime indexed by time then space.
*)

Definition space_time := nat -> row.

(*
  Rule 30 local update.
*)

Definition rule30 (a b c : cell) : cell :=
  xorb a (orb b c).

(*
  One global Rule 30 update on rows.
*)

Definition step_row (r : row) : row :=
  fun x => rule30 (r (x - 1)) (r x) (r (x + 1)).

(*
  Iteration of the global Rule 30 update.
*)

Fixpoint iter_row (n : nat) (r : row) : row :=
  match n with
  | 0%nat => r
  | S n' => step_row (iter_row n' r)
  end.

(*
  Membership in the radius-n interval around x.
*)

Definition in_interval (x : Z) (n : nat) (y : Z) : Prop :=
  x - Z.of_nat n <= y <= x + Z.of_nat n.

(*
  Agreement of two rows on the interval [x-n, x+n].
*)

Definition same_on_interval (u v : row) (x : Z) (n : nat) : Prop :=
  forall y, in_interval x n y -> u y = v y.

(*
  Spatial shift of a row by offset k.
*)

Definition shift_row (k : Z) (r : row) : row :=
  fun x => r (x - k).

(*
  One-step evolution law.
*)

Definition evolves (u : space_time) : Prop :=
  forall t x,
    u (S t) x = rule30 (u t (x - 1)) (u t x) (u t (x + 1)).

End Base_Definitions.

Section Base_Properties.

(*
  Exhaustive truth table for Rule 30.
*)

Theorem rule30_truth_table :
  forall a b c,
    rule30 a b c =
      match a, b, c with
      | false, false, false => false
      | false, false, true => true
      | false, true, false => true
      | false, true, true => true
      | true, false, false => true
      | true, false, true => false
      | true, true, false => false
      | true, true, true => false
      end.
Proof.
  intros a b c.
  destruct a, b, c; reflexivity.
Qed.

(*
  Interval agreement can be restricted to any smaller radius.
*)

Lemma same_on_interval_weaken :
  forall u v x n m,
    (m <= n)%nat ->
    same_on_interval u v x n ->
    same_on_interval u v x m.
Proof.
  intros u v x n m Hmn Hsame y Hy.
  apply Hsame.
  unfold in_interval in *.
  assert (Hmn_z : (Z.of_nat m <= Z.of_nat n)%Z).
  { apply Nat2Z.inj_le. exact Hmn. }
  lia.
Qed.

(*
  Radius (S n) agreement implies radius n agreement at the same center.
*)

Lemma same_on_interval_shift_center :
  forall u v x n,
    same_on_interval u v x (S n) ->
    same_on_interval u v x n.
Proof.
  intros u v x n Hsame.
  eapply same_on_interval_weaken.
  - apply Nat.le_succ_diag_r.
  - exact Hsame.
Qed.

(*
  Radius (S n) agreement around x implies radius n agreement around x-1.
*)

Lemma same_on_interval_shift_left :
  forall u v x n,
    same_on_interval u v x (S n) ->
    same_on_interval u v (x - 1) n.
Proof.
  intros u v x n Hsame y Hy.
  apply Hsame.
  unfold in_interval in *.
  rewrite Nat2Z.inj_succ.
  lia.
Qed.

(*
  Radius (S n) agreement around x implies radius n agreement around x+1.
*)

Lemma same_on_interval_shift_right :
  forall u v x n,
    same_on_interval u v x (S n) ->
    same_on_interval u v (x + 1) n.
Proof.
  intros u v x n Hsame y Hy.
  apply Hsame.
  unfold in_interval in *.
  rewrite Nat2Z.inj_succ.
  lia.
Qed.

(*
  One Rule 30 step shrinks the required agreement radius by one.
*)

Lemma step_row_locality :
  forall u v x n,
    same_on_interval u v x (S n) ->
    same_on_interval (step_row u) (step_row v) x n.
Proof.
  intros u v x n Hsame y Hy.
  unfold step_row.
  assert (Hleft : u (y - 1) = v (y - 1)).
  { apply Hsame.
    unfold in_interval in *.
    rewrite Nat2Z.inj_succ.
    lia. }
  assert (Hmid : u y = v y).
  { apply Hsame.
    unfold in_interval in *.
    rewrite Nat2Z.inj_succ.
    lia. }
  assert (Hright : u (y + 1) = v (y + 1)).
  { apply Hsame.
    unfold in_interval in *.
    rewrite Nat2Z.inj_succ.
    lia. }
  rewrite Hleft, Hmid, Hright.
  reflexivity.
Qed.

(*
  Finite-speed dependence: after n steps, position x depends only on
  initial values in [x-n, x+n].
*)

Theorem iter_row_locality :
  forall n x u v,
    same_on_interval u v x n ->
    iter_row n u x = iter_row n v x.
Proof.
  induction n as [|n IH]; intros x u v Hsame.
  - simpl.
    apply Hsame.
    unfold in_interval.
    simpl.
    lia.
  - simpl.
    assert (Hleft : iter_row n u (x - 1) = iter_row n v (x - 1)).
    { apply IH.
      apply same_on_interval_shift_left.
      exact Hsame. }
    assert (Hmid : iter_row n u x = iter_row n v x).
    { apply IH.
      apply same_on_interval_shift_center.
      exact Hsame. }
    assert (Hright : iter_row n u (x + 1) = iter_row n v (x + 1)).
    { apply IH.
      apply same_on_interval_shift_right.
      exact Hsame. }
    unfold step_row.
    rewrite Hleft, Hmid, Hright.
    reflexivity.
Qed.

(*
  One-step update commutes with spatial shifts, pointwise.
*)

Lemma step_row_shift_value :
  forall r k x,
    step_row (shift_row k r) x = shift_row k (step_row r) x.
Proof.
  intros r k x.
  unfold step_row, shift_row.
  replace ((x - 1) - k) with ((x - k) - 1) by lia.
  replace ((x + 1) - k) with ((x - k) + 1) by lia.
  reflexivity.
Qed.

(*
  Iterated updates commute with spatial shifts, pointwise.
*)

Lemma iter_row_shift_value :
  forall n r k x,
    iter_row n (shift_row k r) x = shift_row k (iter_row n r) x.
Proof.
  induction n as [|n IH]; intros r k x.
  - simpl. reflexivity.
  - simpl.
    unfold step_row.
    rewrite (IH r k (x - 1)).
    rewrite (IH r k x).
    rewrite (IH r k (x + 1)).
    unfold shift_row.
    replace ((x - 1) - k) with ((x - k) - 1) by lia.
    replace ((x + 1) - k) with ((x - k) + 1) by lia.
    reflexivity.
Qed.

End Base_Properties.

Section Seed_Definitions.

(*
  Single-seed initial row: true exactly at coordinate 0.
*)

Definition seed_row : row :=
  fun x => Z.eqb x 0%Z.

(*
  Shifted single-seed row, active at coordinate c.
*)

Definition shifted_seed_row (c : Z) : row :=
  shift_row c seed_row.

(*
  A trajectory is seeded when time 0 matches seed_row and the
  Rule 30 recurrence holds.
*)

Definition seeded_trajectory (u : space_time) : Prop :=
  u 0%nat = seed_row /\ evolves u.

End Seed_Definitions.

Section Seed_Properties.

(*
  Seed center is active.
*)

Lemma seed_row_at_zero :
  seed_row 0%Z = true.
Proof. reflexivity. Qed.

(*
  Off-center seed cells are inactive.
*)

Lemma seed_row_neq_zero :
  forall x,
    x <> 0%Z ->
    seed_row x = false.
Proof.
  intros x Hx.
  unfold seed_row.
  apply Z.eqb_neq.
  exact Hx.
Qed.

(*
  Characterization of seed_row values.
*)

Lemma seed_row_spec :
  forall x,
    seed_row x = true <-> x = 0%Z.
Proof.
  intro x.
  unfold seed_row.
  rewrite Z.eqb_eq.
  tauto.
Qed.

(*
  Shifted seed row is active at its center.
*)

Lemma shifted_seed_row_at_center :
  forall c,
    shifted_seed_row c c = true.
Proof.
  intro c.
  unfold shifted_seed_row, shift_row.
  replace (c - c)%Z with 0%Z by lia.
  apply seed_row_at_zero.
Qed.

(*
  Shifted seed row is inactive away from its center.
*)

Lemma shifted_seed_row_off_center :
  forall c x,
    x <> c ->
    shifted_seed_row c x = false.
Proof.
  intros c x Hxc.
  unfold shifted_seed_row, shift_row.
  apply seed_row_neq_zero.
  lia.
Qed.

End Seed_Properties.

(*

Lemma rule30_111 :

Proof. reflexivity. Qed.

(*
  Operational predecessor-output predicates.
*)

Definition outputs_true (a b c : bool) : Prop :=
  rule30 a b c = true.

Definition outputs_false (a b c : bool) : Prop :=
  rule30 a b c = false.

(*
  Exact predecessor triples yielding output true.
*)

Lemma rule30_eq_true_iff :
  forall a b c,
    rule30 a b c = true <->
      (a = false /\ b = false /\ c = true) \/
      (a = false /\ b = true /\ c = false) \/
      (a = false /\ b = true /\ c = true) \/
      (a = true /\ b = false /\ c = false).
Proof.
  intros a b c.
  destruct a, b, c; simpl; firstorder congruence.
Qed.

(*
  Direct introduction lemmas for true-output triples.
*)

Lemma rule30_true_intro_001 :
  rule30 false false true = true.
Proof. reflexivity. Qed.

Lemma rule30_true_intro_010 :

Proof. reflexivity. Qed.

Lemma rule30_true_intro_011 :
  rule30 false true true = true.
Proof. reflexivity. Qed.

Lemma rule30_true_intro_100 :
  rule30 true false false = true.
Proof. reflexivity. Qed.

(*
  Exact predecessor triples yielding output false.

Lemma rule30_eq_false_iff :
  forall a b c,
    rule30 a b c = false <->
      (a = false /\ b = false /\ c = false) \/
      (a = true /\ b = false /\ c = true) \/
      (a = true /\ b = true /\ c = false) \/
      (a = true /\ b = true /\ c = true).
Proof.
  intros a b c.
  destruct a, b, c; simpl; firstorder congruence.
Qed.

(*
  Direct introduction lemmas for false-output triples.
*)

Lemma rule30_false_intro_000 :
  rule30 false false false = false.
Proof. reflexivity. Qed.

Lemma rule30_false_intro_101 :
  rule30 true false true = false.
Proof. reflexivity. Qed.

Lemma rule30_false_intro_110 :
  rule30 true true false = false.
Proof. reflexivity. Qed.

Lemma rule30_false_intro_111 :
  rule30 true true true = false.
Proof. reflexivity. Qed.

(*
  A predecessor triple 111 cannot produce a true center cell.
*)

Theorem center_block_above_seed_not_111 :
  forall (r : row),
    rule30 (r (-1)%Z) (r 0%Z) (r 1%Z) = true ->
    ~(r (-1)%Z = true /\ r 0%Z = true /\ r 1%Z = true).
Proof.
  intros r Hcenter [Hleft [Hmid Hright]].
  rewrite Hleft, Hmid, Hright in Hcenter.
  rewrite rule30_111 in Hcenter.
  discriminate.
Qed.

(*
  Seeded trajectories cannot identify the seed center value with
  rule30 111.
*)

Theorem seed_center_not_rule30_111 :
  forall (u : space_time),
    seeded_trajectory u ->
    ~(u 0%nat 0%Z = true /\ u 0%nat 0%Z = rule30 true true true).
Proof.
  intros u _ [Hcenter Hr].
  rewrite rule30_111 in Hr.
  rewrite Hr in Hcenter.
  discriminate.
Qed.

(*
  If a row matches a shifted isolated seed on the full dependency cone
  of depth n for position x, then its n-step value at x equals the
  shifted seed evolution.
*)

Theorem isolated_shifted_seed_cone :
  forall n x c r,
    same_on_interval r (shifted_seed_row c) x n ->
    iter_row n r x = iter_row n seed_row (x - c).
Proof.
  intros n x c r Hagree.
  transitivity (iter_row n (shifted_seed_row c) x).
  - apply iter_row_locality.
    exact Hagree.
  - unfold shifted_seed_row.
    rewrite (iter_row_shift_value n seed_row c x).
    unfold shift_row.
    reflexivity.
Qed.

(*
  Centered form of the shifted-seed cone lemma.
*)

Corollary isolated_shifted_seed_cone_centered :
  forall n z c r,
    same_on_interval r (shifted_seed_row c) (c + z) n ->
    iter_row n r (c + z) = iter_row n seed_row z.
Proof.
  intros n z c r Hagree.
  rewrite (isolated_shifted_seed_cone n (c + z) c r Hagree).
  replace ((c + z) - c)%Z with z by lia.
  reflexivity.
Qed.

End Local_Lemmas.

(*
  Integer segment [start, start + len - 1].
*)

Fixpoint z_segment (start : Z) (len : nat) : list Z :=
  match len with
  | 0%nat => []
  | S len' => start :: z_segment (start + 1) len'
  end.

(*
  Coordinates from -m to m.
*)

Definition centered_coords (m : nat) : list Z :=
  z_segment (- Z.of_nat m) (S (2 * m)).

(*
  Width-(2m+1) centered strip cut from a row.
*)

Definition centered_strip_row (r : row) (m : nat) : list cell :=
  map r (centered_coords m).

(*
  Width-(2m+1) centered strip at time t.
*)

Definition centered_strip (u : space_time) (t m : nat) : list cell :=
  centered_strip_row (u t) m.

(*
  Eventual pure periodicity for centered strips.
*)

Definition eventually_periodic_strip (u : space_time) (m : nat) : Prop :=
  exists T P : nat,
    (0 < P)%nat /\
    forall t,
      (T <= t)%nat ->
      centered_strip u (t + P) m = centered_strip u t m.

Section Witness_Definitions.

(*
  Periodicity assumption for the canonical center column with fixed
  transient T and period P.
*)

Definition center_period_assumption (T P : nat) : Prop :=
  (0 < P)%nat /\
  forall t,
    (T <= t)%nat ->
    center_bit (t + P) = center_bit t.

(*
  Minimal finite witness data for a fixed center-periodicity branch:
  center-column values and predecessor triples above center.
*)

Record center_period_witness : Type := {
  cpw_transient : nat;
  cpw_period : nat;
  cpw_center_bits : list cell;
  cpw_pred_triples : list (cell * cell * cell)
}.

(*
  Center bit lookup in the finite witness.
*)

Definition cpw_center_value (w : center_period_witness) (t : nat) : cell :=
  nth t (cpw_center_bits w) false.

(*
  Predecessor triple lookup in the finite witness.
*)

Definition cpw_pred_triple (w : center_period_witness) (t : nat) :
  cell * cell * cell :=
  nth t (cpw_pred_triples w) (false, false, false).

(*
  Triple projections.
*)

Definition cpw_pred_left (w : center_period_witness) (t : nat) : cell :=
  let '(a, _, _) := cpw_pred_triple w t in a.

Definition cpw_pred_mid (w : center_period_witness) (t : nat) : cell :=
  let '(_, b, _) := cpw_pred_triple w t in b.

Definition cpw_pred_right (w : center_period_witness) (t : nat) : cell :=
  let '(_, _, c) := cpw_pred_triple w t in c.

(*
  Finite-size side conditions for the branch window.
*)

Definition cpw_window_complete (w : center_period_witness) : Prop :=
  (0 < cpw_period w)%nat /\
  (cpw_transient w + cpw_period w < length (cpw_center_bits w))%nat /\
  (cpw_transient w + cpw_period w <= length (cpw_pred_triples w))%nat.

(*
  Center-column periodic closure inside the finite window.
*)

Definition cpw_periodicity_constraints (w : center_period_witness) : Prop :=
  forall k,
    (k < cpw_period w)%nat ->
    cpw_center_value w (cpw_transient w + k) =
      cpw_center_value w (cpw_transient w + cpw_period w + k).

(*
  The branch starts from the true seed center value at time 0.
*)

Definition cpw_seed_constraint (w : center_period_witness) : Prop :=
  cpw_center_value w 0%nat = true.

(*
  Local Rule 30 center update constraints in the finite trace.
*)

Definition cpw_local_center_constraints (w : center_period_witness) : Prop :=
  forall t,
    (S t < length (cpw_center_bits w))%nat ->
    (t < length (cpw_pred_triples w))%nat ->
    cpw_center_value w (S t) =
      rule30
        (cpw_pred_left w t)
        (cpw_pred_mid w t)
        (cpw_pred_right w t).

(*
  Combined center-specific witness constraints.
*)

Definition center_witness_constraints (w : center_period_witness) : Prop :=
  cpw_window_complete w /\
  cpw_periodicity_constraints w /\
  cpw_seed_constraint w /\
  cpw_local_center_constraints w.

(*
  Existence form for finite center-period branches.
*)

Definition center_witness_satisfiable : Prop :=
  exists w, center_witness_constraints w.

(*
  Explicit fixed-(T,P) branch form.
*)

Definition fixed_center_period_witness_exists (T P : nat) : Prop :=
  exists w,
    cpw_transient w = T /\
    cpw_period w = P /\
    center_witness_constraints w.

(*
  Finite-instance extraction from a fixed center-period assumption.
*)

Lemma center_period_assumption_gives_instance :
  forall T P n,
    center_period_assumption T P ->
    center_period_instance T P n.
Proof.
  intros T P n [HP Hper].
  split.
  - exact HP.
  - intros j Hj.
    apply Hper.
    lia.
Qed.

(*
  Generic period-1 contradiction schema: if center bits flip at T->T+1,
  period-1 closure from T is impossible.
*)

Lemma no_center_period_assumption_if_adjacent_flip :
  forall T,
    center_bit (S T) <> center_bit T ->
    ~ center_period_assumption T 1%nat.
Proof.
  intros T Hflip [_ Hper].
  assert (Hle : (T <= T)%nat) by lia.
  specialize (Hper T Hle).
  replace (T + 1)%nat with (S T) in Hper by lia.
  contradiction.
Qed.

Section Fixed_Cone_Definitions.

(*
  Finite backward center cone for fixed transient T and period P:
  center values on k = 0..P and predecessor triples on k = 0..P-1.
*)

Record fixed_center_cone (T P : nat) : Type := {
  fcc_center_bits : list cell;
  fcc_pred_triples : list (cell * cell * cell);
  fcc_center_len : length fcc_center_bits = S P;
  fcc_pred_len : length fcc_pred_triples = P
}.

Arguments fcc_center_bits {T P} _.
Arguments fcc_pred_triples {T P} _.
Arguments fcc_center_len {T P} _.
Arguments fcc_pred_len {T P} _.

(*
  Center value lookup in the fixed cone.
*)

Definition fcc_center_value {T P : nat}
  (cone : fixed_center_cone T P) (k : nat) : cell :=
  nth k (fcc_center_bits cone) false.

(*
  Predecessor triple lookup in the fixed cone.
*)

Definition fcc_pred_triple {T P : nat}
  (cone : fixed_center_cone T P) (k : nat) : cell * cell * cell :=
  nth k (fcc_pred_triples cone) (false, false, false).

(*
  Triple projections for the fixed cone.
*)

Definition fcc_pred_left {T P : nat}
  (cone : fixed_center_cone T P) (k : nat) : cell :=
  let '(a, _, _) := fcc_pred_triple cone k in a.

Definition fcc_pred_mid {T P : nat}
  (cone : fixed_center_cone T P) (k : nat) : cell :=
  let '(_, b, _) := fcc_pred_triple cone k in b.

Definition fcc_pred_right {T P : nat}
  (cone : fixed_center_cone T P) (k : nat) : cell :=
  let '(_, _, c) := fcc_pred_triple cone k in c.

(*
  One-step center constraints across the finite cone.
*)

Definition fcc_local_constraints {T P : nat}
  (cone : fixed_center_cone T P) : Prop :=
  forall k,
    (k < P)%nat ->
    fcc_center_value cone (S k) =
      rule30
        (fcc_pred_left cone k)
        (fcc_pred_mid cone k)
        (fcc_pred_right cone k).

(*
  Period closure in the fixed cone (relative index form).
*)

Definition fcc_periodic_closure {T P : nat}
  (cone : fixed_center_cone T P) : Prop :=
  fcc_center_value cone P = fcc_center_value cone 0%nat.

End Fixed_Cone_Definitions.

Section Period_Two_Overlap.

(*
  Minimal row-T fragment around center for period-2 local branch analysis.
*)

Record period_two_overlap_witness : Type := {
  p2_xm2 : cell;
  p2_xm1 : cell;
  p2_x0 : cell;
  p2_x1 : cell;
  p2_x2 : cell
}.

(*
  Center value at time T.
*)

Definition p2_center_t (w : period_two_overlap_witness) : cell :=
  p2_x0 w.

(*
  Center predecessor triple at time T (for center at T+1).
*)

Definition p2_first_triple (w : period_two_overlap_witness) :
  cell * cell * cell :=
  (p2_xm1 w, p2_x0 w, p2_x1 w).

(*
  Center value at time T+1.
*)

Definition p2_center_t1 (w : period_two_overlap_witness) : cell :=
  rule30 (p2_xm1 w) (p2_x0 w) (p2_x1 w).

(*
  Neighbor values at time T+1 around center.
*)

Definition p2_t1_left (w : period_two_overlap_witness) : cell :=
  rule30 (p2_xm2 w) (p2_xm1 w) (p2_x0 w).

Definition p2_t1_right (w : period_two_overlap_witness) : cell :=
  rule30 (p2_x0 w) (p2_x1 w) (p2_x2 w).

(*
  Center predecessor triple at time T+1 (for center at T+2).
*)

Definition p2_second_triple (w : period_two_overlap_witness) :
  cell * cell * cell :=
  (p2_t1_left w, p2_center_t1 w, p2_t1_right w).

(*
  Center value at time T+2.
*)

Definition p2_center_t2 (w : period_two_overlap_witness) : cell :=
  rule30 (p2_t1_left w) (p2_center_t1 w) (p2_t1_right w).

(*
  Pattern 01 with period-2 closure at center: 0,1,0.
*)

Definition p2_center_pattern_010 (w : period_two_overlap_witness) : Prop :=
  p2_center_t w = false /\
  p2_center_t1 w = true /\
  p2_center_t2 w = false.

(*
  Concrete branch tags inside pattern 01.
*)

Definition p2_first_is_001 (w : period_two_overlap_witness) : Prop :=
  p2_first_triple w = (false, false, true).

Definition p2_second_is_110 (w : period_two_overlap_witness) : Prop :=
  p2_second_triple w = (true, true, false).

(*
  Nontrivial finite branch impossibility via overlap:
  inside pattern 01, branch (001 -> 110) cannot occur.
*)

Theorem period_two_01_branch_001_110_impossible :
  forall w,
    p2_center_pattern_010 w ->
    p2_first_is_001 w ->
    p2_second_is_110 w ->
    False.
Proof.
  intros w [Hc0 [Hc1 Hc2]] Hfirst Hsecond.
  unfold p2_first_is_001, p2_second_is_110, p2_first_triple, p2_second_triple in *.
  unfold p2_t1_left, p2_center_t1, p2_t1_right in *.
  destruct w as [xm2 xm1 x0 x1 x2].
  simpl in *.
  inversion Hfirst; subst.
  inversion Hsecond; subst.
Qed.

End Period_Two_Overlap.

Section Center_Period2_Witness.

(*
  Explicit 4-row local witness for period-2 center analysis.
  Time layers are:
    row A: t
    row B: t+1
    row C: t+2
    row D: t+3 (center only)
*)

(*
  Three-row period-2 witness: rows A (time t), B (time t+1), C (time t+2),
  and center at time t+3, together with all Rule 30 consistency proofs.
*)

Record center_period2_witness : Type := {
  cp2_a_m2 : cell;
  cp2_a_m1 : cell;
  cp2_a_0 : cell;
  cp2_a_1 : cell;
  cp2_a_2 : cell;

  cp2_b_m2 : cell;
  cp2_b_m1 : cell;
  cp2_b_0 : cell;
  cp2_b_1 : cell;
  cp2_b_2 : cell;

  cp2_c_m1 : cell;
  cp2_c_0 : cell;
  cp2_c_1 : cell;

  cp2_d_0 : cell;

  cp2_ab_left : rule30 cp2_a_m2 cp2_a_m1 cp2_a_0 = cp2_b_m1;
  cp2_ab_mid : rule30 cp2_a_m1 cp2_a_0 cp2_a_1 = cp2_b_0;
  cp2_ab_right : rule30 cp2_a_0 cp2_a_1 cp2_a_2 = cp2_b_1;

  cp2_bc_left : rule30 cp2_b_m2 cp2_b_m1 cp2_b_0 = cp2_c_m1;
  cp2_bc_mid : rule30 cp2_b_m1 cp2_b_0 cp2_b_1 = cp2_c_0;
  cp2_bc_right : rule30 cp2_b_0 cp2_b_1 cp2_b_2 = cp2_c_1;

  cp2_cd_mid : rule30 cp2_c_m1 cp2_c_0 cp2_c_1 = cp2_d_0
}.

(*
  Center values for the three consecutive rows used in period-2 closure.
*)

Definition cp2_center_t (W : center_period2_witness) : cell := cp2_b_0 W.
Definition cp2_center_t1 (W : center_period2_witness) : cell := cp2_c_0 W.
Definition cp2_center_t2 (W : center_period2_witness) : cell := cp2_d_0 W.

(*
  Period-2 closure predicate: center at t+2 equals center at t.
*)

Definition cp2_period2_closure (W : center_period2_witness) : Prop :=
  cp2_center_t2 W = cp2_center_t W.

(*
  Pattern 010: center values false, true, false at times t, t+1, t+2.
*)

Definition cp2_center_pattern_010 (W : center_period2_witness) : Prop :=
  cp2_center_t W = false /\
  cp2_center_t1 W = true /\
  cp2_period2_closure W.

(*
  Row-B (time t+1) center-neighbor pattern predicates.
*)

Definition cp2_mid_is_001 (W : center_period2_witness) : Prop :=
  cp2_b_m1 W = false /\
  cp2_b_0 W = false /\
  cp2_b_1 W = true.

Definition cp2_mid_is_100 (W : center_period2_witness) : Prop :=
  cp2_b_m1 W = true /\
  cp2_b_0 W = false /\
  cp2_b_1 W = false.

(*
  Row-C (time t+2) center-neighbor pattern predicates.
*)

Definition cp2_top_is_110 (W : center_period2_witness) : Prop :=
  cp2_c_m1 W = true /\
  cp2_c_0 W = true /\
  cp2_c_1 W = false.

Definition cp2_top_is_111 (W : center_period2_witness) : Prop :=
  cp2_c_m1 W = true /\
  cp2_c_0 W = true /\
  cp2_c_1 W = true.

(*
  Canonical explicit witness at root row t.
*)

Definition canonical_cp2_witness (t : nat) : center_period2_witness.
Proof.
  refine {|
    cp2_a_m2 := canonical_row t (-2)%Z;
    cp2_a_m1 := canonical_row t (-1)%Z;
    cp2_a_0 := canonical_row t 0%Z;
    cp2_a_1 := canonical_row t 1%Z;
    cp2_a_2 := canonical_row t 2%Z;

    cp2_b_m2 := canonical_row (S t) (-2)%Z;
    cp2_b_m1 := canonical_row (S t) (-1)%Z;
    cp2_b_0 := canonical_row (S t) 0%Z;
    cp2_b_1 := canonical_row (S t) 1%Z;
    cp2_b_2 := canonical_row (S t) 2%Z;

    cp2_c_m1 := canonical_row (t + 2) (-1)%Z;
    cp2_c_0 := canonical_row (t + 2) 0%Z;
    cp2_c_1 := canonical_row (t + 2) 1%Z;

    cp2_d_0 := canonical_row (t + 3) 0%Z;

    cp2_ab_left := _;
    cp2_ab_mid := _;
    cp2_ab_right := _;
    cp2_bc_left := _;
    cp2_bc_mid := _;
    cp2_bc_right := _;
    cp2_cd_mid := _
  |}.
  - unfold canonical_row, canonical_trajectory.
    simpl.
    reflexivity.
  - unfold canonical_row, canonical_trajectory.
    simpl.
    reflexivity.
  - unfold canonical_row, canonical_trajectory.
    simpl.
    reflexivity.
  - replace (t + 2)%nat with (S (S t)) by lia.
    unfold canonical_row, canonical_trajectory.
    simpl.
    reflexivity.
  - replace (t + 2)%nat with (S (S t)) by lia.
    unfold canonical_row, canonical_trajectory.
    simpl.
    reflexivity.
  - replace (t + 2)%nat with (S (S t)) by lia.
    unfold canonical_row, canonical_trajectory.
    simpl.
    reflexivity.
  - replace (t + 2)%nat with (S (S t)) by lia.
    replace (t + 3)%nat with (S (S (S t))) by lia.
    unfold canonical_row, canonical_trajectory.
    simpl.
    reflexivity.
Defined.

(*
  Extract the explicit local witness from the canonical trajectory,
  rooted at row t.
*)

Lemma canonical_center_period2_witness :
  forall t,
    exists W : center_period2_witness,
      cp2_center_t W = center_bit (S t) /\
      cp2_center_t1 W = center_bit (t + 2) /\
      cp2_center_t2 W = center_bit (t + 3).
Proof.
  intro t.
  exists (canonical_cp2_witness t).
  - repeat split; reflexivity.
Qed.

(*
  First concrete bounded branch impossibility in the explicit
  period-2 witness: pattern 010 cannot realize branch
  (center predecessor 001) followed by (top predecessor 110).
*)

Theorem cp2_pattern_010_branch_001_110_impossible :
  forall W,
    cp2_center_pattern_010 W ->
    cp2_mid_is_001 W ->
    cp2_top_is_110 W ->
    False.
Proof.
  intros W [_ [_ _]] [Hb_m1 [Hb_0 Hb_1]] [_ [_ Hc_1]].
  destruct W as
    [a_m2 a_m1 a_0 a_1 a_2
     b_m2 b_m1 b_0 b_1 b_2
     c_m1 c_0 c_1 d_0
     Hab_left Hab_mid Hab_right
     Hbc_left Hbc_mid Hbc_right
     Hcd_mid].
  simpl in *.
  rewrite <- Hbc_right in Hc_1.
  rewrite Hb_0, Hb_1 in Hc_1.
  simpl in Hc_1.
  discriminate.
Qed.

(*
  Under center pattern 010, the middle predecessor triple is constrained
  to one of the two overlap branches 001 or 100.
*)

Lemma cp2_pattern_010_mid_cases :
  forall W,
    cp2_center_pattern_010 W ->
    cp2_mid_is_001 W \/ cp2_mid_is_100 W.
Proof.
  intros W [Hb0 [Hc0 _]].
  unfold cp2_center_t in Hb0.
  unfold cp2_center_t1 in Hc0.
  rewrite <- (cp2_bc_mid W) in Hc0.
  rewrite Hb0 in Hc0.
  destruct (cp2_b_m1 W) eqn:Hb_m1;
  destruct (cp2_b_1 W) eqn:Hb_1;
  simpl in Hc0; try discriminate.
  - right.
    unfold cp2_mid_is_100.
    repeat split; assumption.
  - left.
    unfold cp2_mid_is_001.
    repeat split; assumption.
Qed.

(*
  Under center pattern 010, the top-left predecessor bit is forced to true.
*)

Lemma cp2_pattern_010_forces_top_left_true :
  forall W,
    cp2_center_pattern_010 W ->
    cp2_c_m1 W = true.
Proof.
  intros W [Hb0 [Hc0 Hclose]].
  unfold cp2_period2_closure in Hclose.
  unfold cp2_center_t2 in Hclose.
  unfold cp2_center_t1 in Hc0.
  rewrite Hb0 in Hclose.
  rewrite <- (cp2_cd_mid W) in Hclose.
  rewrite Hc0 in Hclose.
  destruct (cp2_c_m1 W); simpl in Hclose.
  - reflexivity.
  - discriminate.
Qed.

(*
  Mid triple 001 forces the right top predecessor bit to true.
*)

Lemma cp2_mid_001_forces_top_right_true :
  forall W,
    cp2_mid_is_001 W ->
    cp2_c_1 W = true.
Proof.
  intros W [Hb_m1 [Hb_0 Hb_1]].
  rewrite <- (cp2_bc_right W).
  rewrite Hb_0, Hb_1.
  simpl.
  reflexivity.
Qed.

(*
  Mid triple 100 makes the top-right predecessor equal to b_2.
*)

Lemma cp2_mid_100_forces_top_right_eq_b2 :
  forall W,
    cp2_mid_is_100 W ->
    cp2_c_1 W = cp2_b_2 W.
Proof.
  intros W [Hb_m1 [Hb_0 Hb_1]].

    cp2_center_pattern_101 W ->
    cp2_mid_is_111 W ->
    cp2_b_m2 W = false.
Proof.
  intros W Hpat [Hb_m1 [Hb_0 Hb_1]].
  assert (Hneq : cp2_c_m1 W <> cp2_c_1 W).
  { apply cp2_pattern_101_forces_top_sides_neq.
    exact Hpat. }
  assert (Hc1_false : cp2_c_1 W = false).
  {
    rewrite <- (cp2_bc_right W).
    rewrite Hb_0, Hb_1.
    unfold rule30.
    simpl.
    reflexivity.
  }
  assert (Hcm1_true : cp2_c_m1 W = true).
  {
    destruct (cp2_c_m1 W) eqn:Hcm1.
    - reflexivity.
    - exfalso.
      apply Hneq.
      rewrite Hc1_false.
      reflexivity.
  }
  rewrite <- (cp2_bc_left W) in Hcm1_true.
  rewrite Hb_m1, Hb_0 in Hcm1_true.
  unfold rule30 in Hcm1_true.
  simpl in Hcm1_true.
  destruct (cp2_b_m2 W) eqn:Hb_m2.
  - discriminate.
  - reflexivity.
Qed.

(*
  Pattern 101 + mid 110 forces b_m2 and b_2 to differ.
*)

Lemma cp2_pattern_101_mid_110_forces_b_m2_neq_b_2 :
  forall W,
    cp2_center_pattern_101 W ->
    cp2_mid_is_110 W ->
    cp2_b_m2 W <> cp2_b_2 W.
Proof.
  intros W Hpat [Hb_m1 [Hb_0 Hb_1]] Heq.
  assert (Hneq : cp2_c_m1 W <> cp2_c_1 W).
  { apply cp2_pattern_101_forces_top_sides_neq.
    exact Hpat. }
  assert (Hcm1 : cp2_c_m1 W = xorb (cp2_b_m2 W) true).
  {
    rewrite <- (cp2_bc_left W).
    rewrite Hb_m1, Hb_0.
    unfold rule30.
    simpl.
    reflexivity.
  }
  assert (Hc1 : cp2_c_1 W = xorb true (cp2_b_2 W)).
  {
    rewrite <- (cp2_bc_right W).
    rewrite Hb_0, Hb_1.
    unfold rule30.
    simpl.
    reflexivity.
  }
  rewrite <- Heq in Hc1.
  rewrite xorb_comm in Hc1.
  assert (Heq_top : cp2_c_m1 W = cp2_c_1 W).
  {
    rewrite Hcm1.
    rewrite <- Hc1.
    reflexivity.
  }
  apply Hneq.
  exact Heq_top.
Qed.

(*
  Normal form for pattern 101 at the middle row.
*)

Lemma cp2_pattern_101_normal_form :
  forall W,
    cp2_center_pattern_101 W ->
    (cp2_b_m1 W = true /\ cp2_b_1 W = false /\ cp2_b_m2 W <> cp2_b_2 W) \/
    (cp2_b_m1 W = true /\ cp2_b_1 W = true /\ cp2_b_m2 W = false).
Proof.
  intros W Hpat.
  destruct (cp2_pattern_101_mid_cases W Hpat) as [Hmid110 | Hmid111].
  - destruct Hmid110 as [Hb_m1 [Hb_0 Hb_1]].
    pose proof (cp2_pattern_101_mid_110_forces_b_m2_neq_b_2 W Hpat
      (conj Hb_m1 (conj Hb_0 Hb_1))) as Hneq.
    left.
    repeat split; assumption.
  - destruct Hmid111 as [Hb_m1 [Hb_0 Hb_1]].
    pose proof (cp2_pattern_101_mid_111_forces_b_m2_false W Hpat
      (conj Hb_m1 (conj Hb_0 Hb_1))) as Hb_m2_false.
    right.
    repeat split; assumption.
Qed.

Definition cp2_residual_R101_A (W : center_period2_witness) : Prop :=

  cp2_b_m1 W = true /\
  cp2_b_1 W = true /\
  cp2_b_m2 W = false.

Definition cp2_residual_R101_B1 (W : center_period2_witness) : Prop :=
  cp2_center_pattern_101 W /\
  cp2_b_m1 W = true /\
  cp2_b_1 W = false /\
  cp2_b_m2 W = false /\
  cp2_b_2 W = true.

Definition cp2_residual_R101_B2 (W : center_period2_witness) : Prop :=
  cp2_center_pattern_101 W /\
  cp2_b_m1 W = true /\
  cp2_b_1 W = false /\
  cp2_b_m2 W = true /\
  cp2_b_2 W = false.

Lemma cp2_pattern_101_residual_exhaustion :
  forall W,
    cp2_center_pattern_101 W ->
    cp2_residual_R101_A W \/
    cp2_residual_R101_B1 W \/
    cp2_residual_R101_B2 W.
Proof.
  intros W Hpat.
  destruct (cp2_pattern_101_normal_form W Hpat) as
    [[Hb_m1 [Hb_1_false Hneq]] | [Hb_m1 [Hb_1_true Hb_m2_false]]].
  - destruct (cp2_b_m2 W) eqn:Hb_m2.
    + destruct (cp2_b_2 W) eqn:Hb_2.
      * exfalso.
        apply Hneq.
        reflexivity.
      * right.
        right.
        unfold cp2_residual_R101_B2.
        split.
        -- exact Hpat.
        -- repeat split; assumption.
    + destruct (cp2_b_2 W) eqn:Hb_2.
      * right.
        left.
        unfold cp2_residual_R101_B1.
        split.
        -- exact Hpat.
        -- repeat split; assumption.
      * exfalso.
        apply Hneq.
        reflexivity.
  - left.
    unfold cp2_residual_R101_A.
    split.
    + exact Hpat.
    + repeat split; assumption.
Qed.

(*
  Uniform arbitrary-T contradiction for pattern 101:
  b_1 = true with b_m2 = true is impossible.
*)

Theorem no_center_period_two_from_101_if_b1_true_bm2_true :
  forall T,
    center_period_two_from T ->
    canonical_row (S T) 0%Z = true ->
    canonical_row (T + 2) 0%Z = false ->
    canonical_row (S T) 1%Z = true ->
    canonical_row (S T) (-2)%Z = true ->
    False.
Proof.
  intros T Hper2 Hb_0 Hc_0 Hb_1 Hb_m2.
  set (W := canonical_cp2_witness T).
  assert (Hpat : cp2_center_pattern_101 W).
  {
    unfold cp2_center_pattern_101, cp2_period2_closure, W.
    simpl.
    split.
    - exact Hb_0.
    - split.
      + exact Hc_0.
      + change (center_bit (T + 3) = center_bit (S T)).
        replace (T + 3)%nat with (S T + 2)%nat by lia.
        apply Hper2.
        lia.
  }
  assert (Hb_1_w : cp2_b_1 W = true).
  {
    unfold W.
    simpl.
    exact Hb_1.
  }
  pose proof (cp2_pattern_101_b1_true_forces_mid_111 W Hpat Hb_1_w) as Hmid111.
  pose proof (cp2_pattern_101_mid_111_forces_b_m2_false W Hpat Hmid111) as Hb_m2_false.
  unfold W in Hb_m2_false.
  simpl in Hb_m2_false.
  rewrite Hb_m2 in Hb_m2_false.
  discriminate.
Qed.

Lemma cp2_residual_R010_A_forces_next_R101_A :

    cp2_residual_R101_A (canonical_cp2_witness (S T)).

  intros T Hper2 [Hpat [Hb_m1_false [Hb_1_true Hb_m2_true]]].
  destruct Hpat as [Hb_0_false [Hc_0_true Hclose]].
  assert (Hper2_succ : center_period_two_from (S T)).
  {
    apply (center_period_two_from_monotone T (S T)).
    - lia.
    - exact Hper2.
  }
  assert (Hmid001 : cp2_mid_is_001 (canonical_cp2_witness T)).
  {
    unfold cp2_mid_is_001.
    repeat split; assumption.

    (canonical_cp2_witness T)
    (conj Hb_0_false (conj Hc_0_true Hclose))
    Hmid001) as Htop111.
  destruct Htop111 as [Hc_m1_true [_ Hc_1_true]].
  assert (Hrow_T2_0_true : canonical_row (T + 2) 0%Z = true).
  {
    unfold cp2_center_t1 in Hc_0_true.
    simpl in Hc_0_true.
    exact Hc_0_true.

  {
    change (canonical_row (T + 2) 1%Z = true) in Hc_1_true.
    exact Hc_1_true.

  {
    change (canonical_row (T + 2) (-1)%Z = true) in Hc_m1_true.
    exact Hc_m1_true.

  assert (Hd_0_false : canonical_row (T + 3) 0%Z = false).
  {
    unfold cp2_period2_closure, cp2_center_t2, cp2_center_t in Hclose.
    simpl in Hclose.
    rewrite Hclose.
    exact Hb_0_false.
  }
  unfold cp2_residual_R101_A.
  split.
  - unfold cp2_center_pattern_101, cp2_period2_closure.
    split.
    + unfold cp2_center_t.
      simpl.
      replace (S (S T)) with (T + 2)%nat by lia.
      exact Hrow_T2_0_true.
    + split.
      * unfold cp2_center_t1.
        simpl.
        replace (S (T + 2)) with (T + 3)%nat by lia.
        replace (S T + 2)%nat with (T + 3)%nat by lia.

      * unfold cp2_center_t2, cp2_center_t.
        simpl.
        replace (S (T + 3)) with (T + 4)%nat by lia.

          lia.

        replace (S (S T)) with (T + 2)%nat by lia.
        exact Hperiod.
  - split.
    + unfold cp2_b_m1.
      simpl.
      replace (S (S T)) with (T + 2)%nat by lia.
      exact Hrow_T2_m1_true.
    + split.
      * unfold cp2_b_1.
        simpl.
        replace (S (S T)) with (T + 2)%nat by lia.
        exact Hrow_T2_1_true.
      * unfold cp2_b_m2.
        simpl.
        replace (S (S T)) with (T + 2)%nat by lia.
        destruct (canonical_row (T + 2) (-2)%Z) eqn:Hb_m2_next.
        -- exfalso.
           apply (no_center_period_two_from_101_if_b1_true_bm2_true (S T)).
           ++ exact Hper2_succ.
           ++ replace (S (S T)) with (T + 2)%nat by lia.
              exact Hrow_T2_0_true.
           ++ replace (S T + 2)%nat with (T + 3)%nat by lia.
              exact Hd_0_false.
           ++ replace (S (S T)) with (T + 2)%nat by lia.
              exact Hrow_T2_1_true.
           ++ replace (S (S T)) with (T + 2)%nat by lia.

        -- reflexivity.
Qed.

Lemma cp2_residual_R101_A_forces_prev_R010_residual :

  - lia.
  - destruct H101A as [Hpat101 [_ [_ _]]].

      assert (Hperiod : center_bit (S T' + 2) = center_bit (S T')).

        apply Hper2.
        lia.

      unfold cp2_center_pattern_010, cp2_period2_closure.

      - unfold cp2_center_t.
        simpl.
        exact Hcenter_T_false.

        + unfold cp2_center_t1.
          simpl.
          change (center_bit (T' + 2) = true).
          replace (T' + 2)%nat with (S (S T')) by lia.
          unfold cp2_center_t in Hb_0_true.
          simpl in Hb_0_true.
          exact Hb_0_true.
        + unfold cp2_center_t2, cp2_center_t.
          simpl.
          change (center_bit (T' + 3) = center_bit (S T')).
          replace (T' + 3)%nat with (S T' + 2)%nat by lia.
          apply Hper2.
          lia.

    exact (cp2_pattern_010_residual_exhaustion
      (canonical_cp2_witness T') Hpat_prev).
Qed.

Lemma cp2_residual_R010_B_prev_and_R101_A_current_force_prev_b2_true :
  forall T,
    (1 <= T)%nat ->
    cp2_residual_R010_B (canonical_cp2_witness (T - 1)) ->
    cp2_residual_R101_A (canonical_cp2_witness T) ->
    cp2_b_2 (canonical_cp2_witness (T - 1)) = true.
Proof.
  intros T Hge1 H010B_prev H101A_cur.
  destruct T as [|T'].
  - lia.
  - replace (S T' - 1)%nat with T' in * by lia.
    destruct H101A_cur as [_ [_ [Hb_1_true_cur _]]].
    destruct H010B_prev as [Hpat010 [Hb_m1_true_prev [Hb_1_false_prev Hb_m2_false_prev]]].
    destruct Hpat010 as [Hb_0_false_prev [_ _]].
    assert (Hmid100_prev : cp2_mid_is_100 (canonical_cp2_witness T')).
    {
      unfold cp2_mid_is_100.
      repeat split; assumption.
    }
    pose proof (cp2_mid_100_forces_top_right_eq_b2
      (canonical_cp2_witness T') Hmid100_prev) as Hc1_eq_b2_prev.
    assert (Hc1_prev_eq_b1_cur :
      cp2_c_1 (canonical_cp2_witness T') =
      cp2_b_1 (canonical_cp2_witness (S T'))).
    {
      unfold cp2_c_1, cp2_b_1.
      simpl.
      replace (S (S T')) with (T' + 2)%nat by lia.
      reflexivity.
    }
    rewrite <- Hc1_prev_eq_b1_cur in Hb_1_true_cur.
    rewrite Hc1_eq_b2_prev in Hb_1_true_cur.
    exact Hb_1_true_cur.
Qed.

(*
  Residual transition step:
  under period-2 closure, R101-B1 at time T forces R010-B at time S T.
*)

Lemma cp2_residual_R101_B1_forces_next_R010_B :
  forall T,
    center_period_two_from T ->
    cp2_residual_R101_B1 (canonical_cp2_witness T) ->
    cp2_residual_R010_B (canonical_cp2_witness (S T)).
Proof.
  intros T Hper2 [Hpat [Hb_m1_true [Hb_1_false [Hb_m2_false Hb_2_true]]]].
  destruct Hpat as [Hb_0_true [Hc_0_false Hclose]].
  assert (Hb_0_true_b : cp2_b_0 (canonical_cp2_witness T) = true).
  {
    exact Hb_0_true.
  }
  assert (Hper2_succ : center_period_two_from (S T)).
  {
    apply (center_period_two_from_monotone T (S T)).
    - lia.
    - exact Hper2.
  }
  assert (Hrow_T2_0_false : canonical_row (T + 2) 0%Z = false).
  {
    unfold cp2_center_t1 in Hc_0_false.
    simpl in Hc_0_false.
    exact Hc_0_false.
  }
  assert (Hd_0_true : canonical_row (T + 3) 0%Z = true).
  {
    unfold cp2_period2_closure, cp2_center_t2, cp2_center_t in Hclose.
    simpl in Hclose.
    rewrite Hclose.
    exact Hb_0_true.
  }
  assert (Hc_m1_true : cp2_c_m1 (canonical_cp2_witness T) = true).
  {
    rewrite <- (cp2_bc_left (canonical_cp2_witness T)).
    rewrite Hb_m2_false, Hb_m1_true, Hb_0_true_b.
    unfold rule30.
    simpl.
    reflexivity.
  }
  assert (Hc_1_false : cp2_c_1 (canonical_cp2_witness T) = false).
  {
    rewrite <- (cp2_bc_right (canonical_cp2_witness T)).
    rewrite Hb_0_true_b, Hb_1_false, Hb_2_true.
    unfold rule30.
    simpl.
    reflexivity.
  }
  assert (Hrow_T2_m1_true : canonical_row (T + 2) (-1)%Z = true).
  {
    change (canonical_row (T + 2) (-1)%Z = true) in Hc_m1_true.
    exact Hc_m1_true.
  }
  assert (Hrow_T2_1_false : canonical_row (T + 2) 1%Z = false).
  {
    change (canonical_row (T + 2) 1%Z = false) in Hc_1_false.
    exact Hc_1_false.
  }
  unfold cp2_residual_R010_B.
  split.
  - unfold cp2_center_pattern_010, cp2_period2_closure.
    split.
    + unfold cp2_center_t.
      simpl.
      replace (S (S T)) with (T + 2)%nat by lia.
      exact Hrow_T2_0_false.
    + split.
      * unfold cp2_center_t1.
        simpl.
        replace (S (T + 2)) with (T + 3)%nat by lia.
        replace (S T + 2)%nat with (T + 3)%nat by lia.
        exact Hd_0_true.
      * unfold cp2_center_t2, cp2_center_t.
        simpl.
        replace (S (T + 3)) with (T + 4)%nat by lia.
        assert (Hperiod : center_bit (T + 4) = center_bit (T + 2)).
        {
          replace (T + 4)%nat with (T + 2 + 2)%nat by lia.
          apply Hper2.
          lia.
        }
        unfold center_bit in Hperiod.
        replace (S (S T)) with (T + 2)%nat by lia.
        exact Hperiod.
  - split.
    + unfold cp2_b_m1.
      simpl.
      replace (S (S T)) with (T + 2)%nat by lia.
      exact Hrow_T2_m1_true.
    + split.
      * unfold cp2_b_1.
        simpl.
        replace (S (S T)) with (T + 2)%nat by lia.
        exact Hrow_T2_1_false.
      * unfold cp2_b_m2.
        simpl.
        replace (S (S T)) with (T + 2)%nat by lia.
        destruct (canonical_row (T + 2) (-2)%Z) eqn:Hb_m2_next.
        -- exfalso.
           apply (no_center_period_two_from_branch_100_bm2_true (S T)).
           ++ exact Hper2_succ.
           ++ replace (S (S T)) with (T + 2)%nat by lia.
              exact Hb_m2_next.
           ++ replace (S (S T)) with (T + 2)%nat by lia.
              exact Hrow_T2_m1_true.
           ++ replace (S (S T)) with (T + 2)%nat by lia.
              exact Hrow_T2_0_false.
           ++ replace (S (S T)) with (T + 2)%nat by lia.
              exact Hrow_T2_1_false.
           ++ replace (S T + 2)%nat with (T + 3)%nat by lia.
              exact Hd_0_true.
        -- reflexivity.
Qed.

(*
  Finite residual graph abstraction:
  each residual class is a node, and proved transitions induce edges.
*)

Inductive cp2_residual_node : Type :=
| RNode010A
| RNode010B
| RNode101A
| RNode101B1
| RNode101B2.

Definition cp2_residual_node_holds (T : nat) (n : cp2_residual_node) : Prop :=
  match n with
  | RNode010A => cp2_residual_R010_A (canonical_cp2_witness T)
  | RNode010B => cp2_residual_R010_B (canonical_cp2_witness T)
  | RNode101A => cp2_residual_R101_A (canonical_cp2_witness T)
  | RNode101B1 => cp2_residual_R101_B1 (canonical_cp2_witness T)
  | RNode101B2 => cp2_residual_R101_B2 (canonical_cp2_witness T)
  end.

Inductive cp2_residual_phase : Type :=
| Phase010
| Phase101.

Definition cp2_residual_node_phase (n : cp2_residual_node) : cp2_residual_phase :=
  match n with
  | RNode010A => Phase010
  | RNode010B => Phase010
  | RNode101A => Phase101
  | RNode101B1 => Phase101
  | RNode101B2 => Phase101
  end.

Inductive cp2_residual_edge : cp2_residual_node -> cp2_residual_node -> Prop :=
| Edge_010A_101A : cp2_residual_edge RNode010A RNode101A
| Edge_010B_101A : cp2_residual_edge RNode010B RNode101A
| Edge_010B_101B1 : cp2_residual_edge RNode010B RNode101B1
| Edge_010B_101B2 : cp2_residual_edge RNode010B RNode101B2
| Edge_101A_010B : cp2_residual_edge RNode101A RNode010B
| Edge_101B1_010B : cp2_residual_edge RNode101B1 RNode010B
| Edge_101B2_010A : cp2_residual_edge RNode101B2 RNode010A.

Lemma cp2_residual_edge_phase_flip :
  forall n n',
    cp2_residual_edge n n' ->
    cp2_residual_node_phase n <> cp2_residual_node_phase n'.
Proof.
  intros n n' Hedge.
  destruct Hedge; discriminate.
Qed.

Lemma cp2_residual_has_edge_successor :
  forall T n,
    center_period_two_from T ->
    cp2_residual_node_holds T n ->
    exists n',
      cp2_residual_edge n n' /\
      cp2_residual_node_holds (S T) n'.
Proof.
  intros T n Hper2 Hnode.
  destruct n.
  - exists RNode101A.
    split.
    + exact Edge_010A_101A.
    + apply (cp2_residual_R010_A_forces_next_R101_A T Hper2 Hnode).
  - assert (Hper2_succ : center_period_two_from (S T)).
    {
      apply (center_period_two_from_monotone T (S T)).
      - lia.
      - exact Hper2.
    }
    destruct Hnode as [[Hb_0_false [Hc_0_true _]] _].
    change (center_bit (S T) = false) in Hb_0_false.
    change (center_bit (T + 2) = true) in Hc_0_true.
    assert (Hss : center_bit (S (S T)) = true).
    {
      replace (S (S T)) with (T + 2)%nat by lia.
      exact Hc_0_true.
    }
    destruct (canonical_period_two_flip_01_gives_R101_residual
      (S T) Hper2_succ Hb_0_false Hss) as [HA | [HB1 | HB2]].
    + exists RNode101A.
      split.
      * exact Edge_010B_101A.
      * exact HA.
    + exists RNode101B1.
      split.
      * exact Edge_010B_101B1.
      * exact HB1.
    + exists RNode101B2.
      split.
      * exact Edge_010B_101B2.
      * exact HB2.
  - exists RNode010B.
    split.
    + exact Edge_101A_010B.
    + apply (cp2_residual_R101_A_forces_next_R010_B T Hper2 Hnode).
  - exists RNode010B.
    split.
    + exact Edge_101B1_010B.
    + apply (cp2_residual_R101_B1_forces_next_R010_B T Hper2 Hnode).
  - exists RNode010A.
    split.
    + exact Edge_101B2_010A.
    + apply (cp2_residual_R101_B2_forces_next_R010_A T Hper2 Hnode).
Qed.

Lemma cp2_residual_has_phase_flipping_successor :
  forall T n,
    center_period_two_from T ->
    cp2_residual_node_holds T n ->
    exists n',
      cp2_residual_node_holds (S T) n' /\
      cp2_residual_node_phase n' <> cp2_residual_node_phase n.
Proof.
  intros T n Hper2 Hnode.
  destruct (cp2_residual_has_edge_successor T n Hper2 Hnode) as
    [n' [Hedge Hnext]].
  exists n'.
  split.
  - exact Hnext.
  - intro Heq.
    apply (cp2_residual_edge_phase_flip n n' Hedge).
    symmetry.
    exact Heq.
Qed.

(*
  Structural forward facts for the unresolved residual R010-B:
  the next-step middle signature is explicit, and b_2 controls which
  101-residual branch can occur at the next index.
*)

Lemma cp2_residual_R010_B_forces_next_mid_signature :
  forall T,
    cp2_residual_R010_B (canonical_cp2_witness T) ->
    cp2_b_m1 (canonical_cp2_witness (S T)) = true /\
    cp2_b_0 (canonical_cp2_witness (S T)) = true /\
    cp2_b_1 (canonical_cp2_witness (S T)) =
      cp2_b_2 (canonical_cp2_witness T).
Proof.
  intros T [Hpat [Hb_m1_true [Hb_1_false Hb_m2_false]]].
  destruct Hpat as [Hb_0_false [Hc_0_true _]].
  assert (Hb_0_false_b : cp2_b_0 (canonical_cp2_witness T) = false).
  {
    exact Hb_0_false.
  }
  assert (Hmid100 : cp2_mid_is_100 (canonical_cp2_witness T)).
  {
    unfold cp2_mid_is_100.
    repeat split; assumption.
  }
  assert (Hc_m1_true : cp2_c_m1 (canonical_cp2_witness T) = true).
  {
    rewrite <- (cp2_bc_left (canonical_cp2_witness T)).
    rewrite Hb_m2_false, Hb_m1_true, Hb_0_false_b.
    unfold rule30.
    simpl.
    reflexivity.
  }
  assert (Hc_1_eq_b2 :
    cp2_c_1 (canonical_cp2_witness T) = cp2_b_2 (canonical_cp2_witness T)).
  {
    apply (cp2_mid_100_forces_top_right_eq_b2 (canonical_cp2_witness T) Hmid100).
  }
  split.
  - unfold cp2_b_m1.
    simpl.
    replace (S (S T)) with (T + 2)%nat by lia.
    change (cp2_c_m1 (canonical_cp2_witness T) = true).
    exact Hc_m1_true.
  - split.
    + unfold cp2_b_0.
      simpl.
      replace (S (S T)) with (T + 2)%nat by lia.
      exact Hc_0_true.
    + unfold cp2_b_1.
      simpl.
      replace (S (S T)) with (T + 2)%nat by lia.
      change (cp2_c_1 (canonical_cp2_witness T) =
        cp2_b_2 (canonical_cp2_witness T)).
      exact Hc_1_eq_b2.
Qed.

Lemma cp2_residual_R010_B_and_b2_true_forces_next_R101_A :
  forall T,
    center_period_two_from T ->
    cp2_residual_R010_B (canonical_cp2_witness T) ->
    cp2_b_2 (canonical_cp2_witness T) = true ->
    cp2_residual_R101_A (canonical_cp2_witness (S T)).

  destruct (cp2_residual_has_edge_successor T RNode010B Hper2 HR010B) as
    [n' [Hedge Hnext]].
  destruct n'.
  - inversion Hedge.
  - inversion Hedge.
  - exact Hnext.
  - exfalso.
    destruct Hnext as [_ [_ [Hb_1_false [_ _]]]].
    rewrite (proj2 (proj2 (cp2_residual_R010_B_forces_next_mid_signature T HR010B)))
      in Hb_1_false.
    rewrite Hb2_true in Hb_1_false.
    discriminate.
  - exfalso.
    destruct Hnext as [_ [_ [Hb_1_false [_ _]]]].
    rewrite (proj2 (proj2 (cp2_residual_R010_B_forces_next_mid_signature T HR010B)))
      in Hb_1_false.
    rewrite Hb2_true in Hb_1_false.
    discriminate.
Qed.

Lemma cp2_residual_R010_B_and_b2_false_forces_next_R101_B1_or_B2 :
  forall T,
    center_period_two_from T ->
    cp2_residual_R010_B (canonical_cp2_witness T) ->
    cp2_b_2 (canonical_cp2_witness T) = false ->

    rewrite Hb2_false in Hb_1_true.

  - left.
    exact Hnext.
  - right.
    exact Hnext.
Qed.

Lemma cp2_residual_R010_B_b2_true_reduces_to_next_R101_A :
  (forall U,

Qed.

Lemma cp2_residual_R010_B_b2_false_reduces_to_next_R101_B1_or_B2 :

  (forall U,

  - exact (HnextB1 (S T) Hper2_succ H101B1_next).
  - exact (HnextB2 (S T) Hper2_succ H101B2_next).

Lemma cp2_residual_R010_B_closed_if_b2_branches_closed :
  (forall U,
    center_period_two_from U ->
    cp2_residual_R101_A (canonical_cp2_witness U) ->
    False) ->
  (forall U,
    center_period_two_from U ->
    cp2_residual_R101_B1 (canonical_cp2_witness U) ->
    False) ->
  (forall U,
    center_period_two_from U ->
    cp2_residual_R101_B2 (canonical_cp2_witness U) ->
    False) ->
  forall T,
    center_period_two_from T ->
    cp2_residual_R010_B (canonical_cp2_witness T) ->
    False.
Proof.
  intros HnextA HnextB1 HnextB2 T Hper2 HR010B.
  destruct (cp2_b_2 (canonical_cp2_witness T)) eqn:Hb2.
  - apply (cp2_residual_R010_B_b2_true_reduces_to_next_R101_A HnextA

(*
  If period-2 closure holds from T and adjacent bits at T,T+1 agree,
  then the center tail is period-1 from T.
*)

Lemma center_period_two_from_even :
  forall T,
    center_period_two_from T ->
    forall k,
      center_bit (T + 2 * k) = center_bit T.
Proof.
  intros T Hper2 k.
  induction k as [|k IH].
  - simpl.
    replace (T + 0)%nat with T by lia.
    reflexivity.
  - replace (T + 2 * S k)%nat with ((T + 2 * k) + 2)%nat by lia.
    rewrite Hper2.
    + exact IH.
    + lia.
Qed.

Lemma center_period_two_from_odd :
  forall T,
    center_period_two_from T ->
    forall k,
      center_bit (T + (2 * k + 1)) = center_bit (S T).
Proof.
  intros T Hper2 k.
  induction k as [|k IH].
  - replace (T + (2 * 0 + 1))%nat with (S T) by lia.
    reflexivity.
  - replace (T + (2 * S k + 1))%nat with ((T + (2 * k + 1)) + 2)%nat by lia.
    rewrite Hper2.
    + exact IH.
    + lia.
Qed.

Lemma center_period_two_from_adjacent_equal_gives_period_one :
  forall T,
    center_period_two_from T ->
    center_bit (S T) = center_bit T ->
    center_period_assumption T 1%nat.
Proof.
  intros T Hper2 Hadj.
  split.
  - lia.
  - intros t HT.
    set (n := (t - T)%nat).
    assert (Ht : (t = T + n)%nat).
    {
      unfold n.
      lia.
    }
    rewrite Ht.
    assert (Hpair :
      forall m,
        center_bit (T + m + 1) = center_bit (T + m) /\
        center_bit (T + S m + 1) = center_bit (T + S m)).
    {
      intro m.
      induction m as [|m [IHm IHSm]].
      - split.
        + replace (T + 0 + 1)%nat with (S T) by lia.
          replace (T + 0)%nat with T by lia.
          exact Hadj.
        + replace (T + S 0 + 1)%nat with (T + 2)%nat by lia.
          replace (T + S 0)%nat with (S T) by lia.
          rewrite (Hper2 T (le_n _)).
          symmetry.
          exact Hadj.
      - split.
        + exact IHSm.
        + replace (T + S (S m) + 1)%nat with ((T + m + 1) + 2)%nat by lia.
          replace (T + S (S m))%nat with ((T + m) + 2)%nat by lia.
          rewrite (Hper2 (T + m + 1)%nat).
          * rewrite (Hper2 (T + m)%nat).
            -- exact IHm.
            -- lia.
          * lia.
    }
    exact (proj1 (Hpair n)).
Qed.

Theorem no_center_period_two_from_11_if_no_period_one :

Theorem center_period_reduces_to_period_two_or_one_if_large_period_reduction :
  center_period_reduces_large_periods ->
  center_period_reduces_to_period_two_or_one.
Proof.
  intros Hlarge T P [HPpos Hper].
  destruct P as [|P'].
  - lia.
  - destruct P' as [|P''].
    + right.
      split.
      * lia.
      * exact Hper.
    + destruct P'' as [|P'''].
      * left.
        apply (proj1 (center_period_assumption_period_two_iff T)).
        split.
        -- lia.
        -- exact Hper.
      * apply (Hlarge T (S (S (S P''')))).
        -- lia.
        -- split.
           ++ lia.
           ++ exact Hper.
Qed.

(*
  No transient T >= 13 admits a period-1 center-period assumption.
*)

Definition center_period_one_large_tail_closed : Prop :=
  forall T,
    (13 <= T)%nat ->
    ~ center_period_assumption T 1%nat.

(*
  Period 1 implies period 2: if center_bit (t+1) = center_bit t for
  all t >= T, then center_bit (t+2) = center_bit t as well.
*)

Lemma center_period_assumption_one_implies_period_two :
  forall T,
    center_period_assumption T 1%nat ->
    center_period_two_from T.
Proof.
  intros T [_ Hper] t Ht.
  assert (Ht1 : center_bit (t + 1)%nat = center_bit t).
  {
    apply Hper.
    exact Ht.
  }
  assert (Ht2 : center_bit (t + 2)%nat = center_bit (t + 1)%nat).
  {
    replace (t + 2)%nat with ((t + 1) + 1)%nat by lia.
    apply Hper.
    lia.
  }
  rewrite Ht2.
  exact Ht1.
Qed.

  destruct T as [|T]. { exact (no_center_period_two_from_5 Hper2). }
  destruct T as [|T]. { exact (no_center_period_two_from_6 Hper2). }
  destruct T as [|T]. { exact (no_center_period_two_from_7 Hper2). }
  destruct T as [|T]. { exact (no_center_period_two_from_8 Hper2). }

(* R06__No_Strip_Periodicity.v *)

From Coq Require Import Arith Lia List ZArith.
Import ListNotations.

From T004 Require Import
  R03__Periodicity
  R04__Witness
  R05__Refutation.

(*************************************************************************)
(*                                                                       *)
(*  Proofcase / Rule 30 Finite Obstruction — No Strip Periodicity        *)
(*                                                                       *)
(*  Strip-level consequence layer reducing strip periodicity to center   *)
(*  periodicity and refuting it.  We show width-(2m+1) centered strip   *)
(*  periodicity collapses to width-0, which is center periodicity.       *)
(*                                                                       *)
(*************************************************************************)

Section No_Strip_Periodicity.

(*
  Width-0 centered coordinate list is the singleton [0].
*)

Lemma centered_coords_width0 :
  centered_coords 0%nat = [0%Z].
Proof.
  unfold centered_coords.
  simpl.
  reflexivity.
Qed.

(*
  Width-0 centered strip is the singleton center value.
*)

Lemma centered_strip_width0_general :
  forall u t,
    centered_strip u t 0%nat = [u t 0%Z].
Proof.
  intros u t.
  unfold centered_strip, centered_strip_row.
  rewrite centered_coords_width0.
  reflexivity.
Qed.

(*
  Width-0 centered strip for the canonical trajectory.
*)

Lemma centered_strip_width0 :
  forall t,
    centered_strip canonical_trajectory t 0%nat = [center_bit t].
Proof.
  intro t.
  unfold center_bit, canonical_row.
  apply centered_strip_width0_general.
Qed.

(*
  Strip projection from width (S m) to width m:
  keep the centered subsegment after removing one cell from each side.
*)

Definition strip_trim {A : Type} (m : nat) (xs : list A) : list A :=
  firstn (S (2 * m)) (tl xs).

(*
  Tail commutes with map.
*)

Lemma map_tl :
  forall (A B : Type) (f : A -> B) (xs : list A),
    tl (map f xs) = map f (tl xs).
Proof.
  intros A B f [|x xs].
  - reflexivity.
  - reflexivity.
Qed.

(*
  Tail of a non-empty z_segment.
*)

Lemma z_segment_tl :
  forall start len,
    tl (z_segment start (S len)) = z_segment (start + 1)%Z len.
Proof.
  intros start len.
  reflexivity.
Qed.

(*
  firstn over z_segment that is one element longer.
*)

Lemma z_segment_firstn :
  forall start len,
    firstn len (z_segment start (S len)) = z_segment start len.
Proof.
  intros start len.
  revert start.
  induction len as [|len IH]; intro start.
  - reflexivity.
  - simpl.
    change ((start + 1)%Z :: z_segment (start + 1 + 1)%Z len)
      with (z_segment (start + 1)%Z (S len)).
    rewrite (IH (start + 1)%Z).
    reflexivity.
Qed.

(*
  Coordinate-level trim identity between widths (S m) and m.
*)

Lemma centered_coords_trim :
  forall m,
    firstn (S (2 * m)) (tl (centered_coords (S m))) = centered_coords m.
Proof.
  intro m.
  unfold centered_coords.
  rewrite z_segment_tl.
  replace (- Z.of_nat (S m) + 1)%Z with (- Z.of_nat m)%Z by lia.
  replace (2 * S m)%nat with (S (S (2 * m)))%nat by lia.
  rewrite z_segment_firstn.
  reflexivity.
Qed.

(*
  strip_trim commutes with map over coordinates.
*)

Lemma strip_trim_map :
  forall (A B : Type) (f : A -> B) m (xs : list A),
    strip_trim m (map f xs) = map f (firstn (S (2 * m)) (tl xs)).
Proof.
  intros A B f m xs.
  unfold strip_trim.
  rewrite map_tl.
  rewrite firstn_map.
  reflexivity.
Qed.

(*
  One-slice centered strip projection from width (S m) to width m.
*)

Lemma centered_strip_trim :
  forall u t m,
    strip_trim m (centered_strip u t (S m)) = centered_strip u t m.
Proof.
  intros u t m.
  unfold centered_strip, centered_strip_row.
  rewrite strip_trim_map.
  rewrite centered_coords_trim.
  reflexivity.
Qed.

(*
  Eventual strip periodicity is monotone downward in width.
*)

Lemma eventually_periodic_strip_down :
  forall u m,
    eventually_periodic_strip u (S m) ->
    eventually_periodic_strip u m.
Proof.
  intros u m [T [P [HPpos Hperiodic]]].
  exists T, P.
  split.
  - exact HPpos.
  - intros t Ht.
    specialize (Hperiodic t Ht).
    apply (f_equal (strip_trim m)) in Hperiodic.
    rewrite centered_strip_trim in Hperiodic.
    rewrite centered_strip_trim in Hperiodic.
    exact Hperiodic.
Qed.

(*
  Collapse eventual strip periodicity at any width to width 0.
*)

Lemma eventually_periodic_strip_to_width0 :
  forall u m,
    eventually_periodic_strip u m ->
    eventually_periodic_strip u 0%nat.
Proof.
  intros u m.
  induction m as [|m IH]; intro Hstrip.
  - exact Hstrip.
  - apply IH.
    apply eventually_periodic_strip_down.
    exact Hstrip.
Qed.

(*
  Width-0 strip eventual periodicity implies eventual periodicity
  of the center column.
*)

Lemma eventually_periodic_strip0_implies_center :
  eventually_periodic_strip canonical_trajectory 0%nat ->
  eventually_periodic_center.
Proof.
  intros [T [P [HPpos Hstrip]]].
  exists T, P.
  split.
  - exact HPpos.
  - intros t Ht.
    specialize (Hstrip t Ht).
    rewrite centered_strip_width0 in Hstrip.
    rewrite centered_strip_width0 in Hstrip.
    injection Hstrip as Hcenter.
    exact Hcenter.
Qed.

(*
  No centered finite strip of the canonical Rule 30 trajectory is
  eventually periodic.
*)

Theorem no_eventually_periodic_centered_strip :
  phase2_center_core_closed ->
  forall m,
    ~ eventually_periodic_strip canonical_trajectory m.
Proof.
  intros Hcore m Hstrip.
  apply (center_not_eventually_periodic Hcore).
  apply eventually_periodic_strip0_implies_center.
  apply (eventually_periodic_strip_to_width0 canonical_trajectory m).
  exact Hstrip.
Qed.

Theorem no_eventually_periodic_centered_strip_from_backward_two_step_payload :
  phase2_center_core_closed_via_backward_two_step_payload ->
  forall m,
    ~ eventually_periodic_strip canonical_trajectory m.
Proof.
  intros Hcore m Hstrip.
  apply (center_not_eventually_periodic_from_backward_two_step_payload Hcore).
  apply eventually_periodic_strip0_implies_center.
  apply (eventually_periodic_strip_to_width0 canonical_trajectory m).
  exact Hstrip.
Qed.

(*
  Ge3-refutation endpoint with explicit large-T input.
  This keeps the strip-level theorem constructive and assumption-transparent.
*)

Theorem no_eventually_periodic_centered_strip_from_ge3_refutation_core :
  no_center_period_two_from_ge13_prop ->
  phase2_center_core_closed_with_ge3_refutation ->
  forall m,
    ~ eventually_periodic_strip canonical_trajectory m.
Proof.
  intros Hge13_refute Hcore m Hstrip.
  apply (center_not_eventually_periodic_from_ge3_refutation_core
    Hge13_refute Hcore).
  apply eventually_periodic_strip0_implies_center.
  apply (eventually_periodic_strip_to_width0 canonical_trajectory m).
  exact Hstrip.
Qed.

Theorem no_eventually_periodic_centered_strip_from_large_period_reduction_and_ge13_refutation :
  center_period_reduces_large_periods ->
  no_center_period_two_from_ge13_prop ->
  forall m,
    ~ eventually_periodic_strip canonical_trajectory m.
Proof.
  intros Hreduce_large Hge13_refute m Hstrip.
  apply (center_not_eventually_periodic_from_large_period_reduction_and_ge13_refutation
    Hreduce_large Hge13_refute).
  apply eventually_periodic_strip0_implies_center.
  apply (eventually_periodic_strip_to_width0 canonical_trajectory m).
  exact Hstrip.
Qed.

End No_Strip_Periodicity.

(* R07__Packed_Compression.v *)

From Coq Require Import Arith Bool Lia List PeanoNat ZArith.
Import ListNotations.

From T004 Require Import
  R00__Base
  R03__Periodicity.

Open Scope Z_scope.

(*************************************************************************)
(*                                                                       *)
(*  Proofcase / Rule 30 Finite Obstruction — Packed Compression          *)
(*                                                                       *)
(*  Packed-coordinate and compressed-trace interfaces.  We define cone   *)
(*  coordinates, pack/unpack bijections, packed bits and rows, and the   *)
(*  compressed center-line trace.                                        *)
(*                                                                       *)
(*************************************************************************)

Section Packed_Compression.

(*
  Canonical cone coordinates: time index and spatial index.
*)

Definition cone_coord : Type := (nat * Z)%type.

(*
  Packed coordinates: same time index with a nonnegative horizontal index.
*)

Definition packed_coord : Type := (nat * nat)%type.

(*
  Radius-t cone membership around the center.
*)

Definition in_canonical_cone (c : cone_coord) : Prop :=
  let '(t, x) := c in (- Z.of_nat t <= x <= Z.of_nat t)%Z.

(*
  Pack a cone coordinate by shifting x into [0, 2t].
*)

Definition pack_coord (c : cone_coord) : packed_coord :=
  let '(t, x) := c in (t, Z.to_nat (x + Z.of_nat t)).

(*
  Unpack a packed coordinate back to the centered integer axis.
*)

Definition unpack_coord (p : packed_coord) : cone_coord :=
  let '(t, j) := p in (t, Z.of_nat j - Z.of_nat t).

(*
  Unpacking after packing is identity on valid cone coordinates.
*)

Lemma unpack_coord_pack_coord :
  forall t x,
    in_canonical_cone (t, x) ->
    unpack_coord (pack_coord (t, x)) = (t, x).
Proof.
  intros t x Hcone.
  unfold in_canonical_cone in Hcone.
  unfold unpack_coord, pack_coord.
  simpl.
  f_equal.
  assert (Hnonneg : (0 <= x + Z.of_nat t)%Z) by lia.
  rewrite (Z2Nat.id (x + Z.of_nat t) Hnonneg).
  lia.
Qed.

(*
  Packing after unpacking is always identity.
*)

Lemma pack_coord_unpack_coord :
  forall t j,
    pack_coord (unpack_coord (t, j)) = (t, j).
Proof.
  intros t j.
  unfold pack_coord, unpack_coord.
  simpl.
  f_equal.
  replace (Z.of_nat j - Z.of_nat t + Z.of_nat t)%Z with (Z.of_nat j) by lia.
  rewrite Nat2Z.id.
  reflexivity.
Qed.

(*
  Packed horizontal index remains inside [0, 2t] for valid cone points.
*)

Lemma pack_coord_index_bound :
  forall t x,
    in_canonical_cone (t, x) ->
    (snd (pack_coord (t, x)) <= 2 * t)%nat.
Proof.
  intros t x Hcone.
  unfold in_canonical_cone in Hcone.
  unfold pack_coord.
  simpl.
  apply Nat2Z.inj_le.
  assert (Hnonneg : (0 <= x + Z.of_nat t)%Z) by lia.
  rewrite (Z2Nat.id (x + Z.of_nat t) Hnonneg).
  replace (Z.of_nat (2 * t)) with (2 * Z.of_nat t)%Z by lia.
  lia.
Qed.

(*
  Read one packed bit from an arbitrary source row.
*)

Definition packed_bit_of_row (r : row) (t j : nat) : cell :=
  r (Z.of_nat j - Z.of_nat t).

(*
  Packed row of width 2t+1 from an arbitrary source row.
*)

Definition packed_row_of_row (r : row) (t : nat) : list cell :=
  map (packed_bit_of_row r t) (seq 0 (S (2 * t))).

(*
  Extensionality for packed bits under source-row equality.
*)

Lemma packed_bit_of_row_ext :
  forall r1 r2 t j,
    (forall x, r1 x = r2 x) ->
    packed_bit_of_row r1 t j = packed_bit_of_row r2 t j.
Proof.
  intros r1 r2 t j Hrow.
  unfold packed_bit_of_row.
  apply Hrow.
Qed.

(*
  Extensionality for packed rows under source-row equality.
*)

Lemma packed_row_of_row_ext :
  forall r1 r2 t,
    (forall x, r1 x = r2 x) ->
    packed_row_of_row r1 t = packed_row_of_row r2 t.
Proof.
  intros r1 r2 t Hrow.
  unfold packed_row_of_row.
  apply map_ext.
  intro j.
  apply packed_bit_of_row_ext.
  exact Hrow.
Qed.

Definition packed_bit (t j : nat) : cell :=
  packed_bit_of_row (canonical_row t) t j.

Definition packed_row (t : nat) : list cell :=

Definition packed_trace (t : nat) : packed_state :=

(*
  Packed center coordinate (t,0) maps to packed index t.

Lemma pack_coord_center :
  forall t,
    pack_coord (t, 0%Z) = (t, t).
Proof.
  intro t.
  unfold pack_coord.
  simpl.
  replace (0 + Z.of_nat t)%Z with (Z.of_nat t) by lia.
  rewrite Nat2Z.id.
  reflexivity.
Qed.

    compressed_trace t = center_bit t.

  replace (Z.of_nat t - Z.of_nat t)%Z with 0%Z by lia.
  reflexivity.
Qed.

Lemma packed_trace_of_center :
  forall t,
    packed_trace t = center_bit t.
Proof.
  intro t.
  unfold packed_trace.
  apply compressed_trace_of_center.
Qed.

  Packed-row length is 2t+1.

Lemma packed_row_length :
  forall t,
    length (packed_row t) = S (2 * t).
Proof.
  intro t.
  unfold packed_row, packed_row_of_row.
  rewrite length_map.
  rewrite length_seq.
  reflexivity.
Qed.

(*
  The packed center index is always in range.

Lemma packed_center_index_in_bounds :
  forall t,
    (t < length (packed_row t))%nat.
Proof.
  intro t.
  rewrite packed_row_length.
  lia.
Qed.

(*
  Reading a packed row by index returns packed_bit at that index.
*)

Lemma nth_error_packed_row :
  forall t j,
    (j < length (packed_row t))%nat ->
    nth_error (packed_row t) j = Some (packed_bit t j).
Proof.
  intros t j Hj.
  assert (Hj_seq : (j < S (2 * t))%nat).
  { rewrite <- packed_row_length. exact Hj. }
  unfold packed_row, packed_row_of_row, packed_bit.
  rewrite nth_error_map.
  rewrite nth_error_seq.
  match goal with
  | |- context [if ?b then _ else _] => destruct b eqn:Hb
  end.
  - simpl.
    replace (0 + j)%nat with j by lia.
    reflexivity.
  - apply Nat.ltb_ge in Hb.
    lia.
Qed.

(*
  Shift-by-k view for packed-bit indexing.

Lemma packed_bit_shift :
  forall t j k,
    packed_bit t (j + k)%nat =
      canonical_row t ((Z.of_nat j - Z.of_nat t) + Z.of_nat k)%Z.
Proof.
  intros t j k.
  unfold packed_bit, packed_bit_of_row.
  rewrite Nat2Z.inj_add.
  replace (Z.of_nat j + Z.of_nat k - Z.of_nat t)%Z
    with ((Z.of_nat j - Z.of_nat t) + Z.of_nat k)%Z by lia.
  reflexivity.
Qed.

  Explicit finite-window dependence statement:
  the packed center trace is determined by the width-1 centered strip.

Theorem packed_finite_window :
  exists k,
    forall t,
      nth_error (centered_strip canonical_trajectory t k) k =
      Some (packed_trace t).
Proof.
  exists 0%nat.
  intro t.
  unfold centered_strip, centered_strip_row, centered_coords.
  unfold packed_trace, compressed_trace, packed_bit, packed_bit_of_row.
  simpl.
  replace (Z.of_nat t - Z.of_nat t)%Z with 0%Z by lia.
  reflexivity.
Qed.

(*
  Packed reading from a packed coordinate recovers canonical data.

Lemma packed_bit_from_pack_coord :
  forall t x,
    in_canonical_cone (t, x) ->
    packed_bit t (snd (pack_coord (t, x))) = canonical_row t x.
Proof.
  intros t x Hcone.
  unfold packed_bit, packed_bit_of_row.
  unfold pack_coord.
  simpl.
  assert (Hnonneg : (0 <= x + Z.of_nat t)%Z).
  { unfold in_canonical_cone in Hcone. lia. }
  rewrite (Z2Nat.id (x + Z.of_nat t) Hnonneg).
  replace (x + Z.of_nat t - Z.of_nat t)%Z with x by lia.
  reflexivity.
Qed.

Lemma packed_bit_well_defined_from_canonical :

  exists (snd (pack_coord (t, x))).
  split.
  - apply pack_coord_index_bound.
    exact Hcone.
  - apply packed_bit_from_pack_coord.
    exact Hcone.
Qed.

(*
  Eventual pure periodicity for a nat-indexed sequence.
*)

Definition eventually_periodic_sequence {A : Type} (f : nat -> A) : Prop :=
  exists T P : nat,
    (0 < P)%nat /\
    forall t,
      (T <= t)%nat ->
      f (t + P)%nat = f t.

(*
  Eventual constancy for a nat-indexed sequence.
*)

Definition eventually_constant_sequence {A : Type} (f : nat -> A) : Prop :=
  exists T : nat,
    exists a : A,
      forall t,
        (T <= t)%nat ->
        f t = a.

(*
  Eventual constancy implies eventual periodicity (period 1).
*)

Lemma eventually_constant_implies_eventually_periodic :
  forall (A : Type) (f : nat -> A),
    eventually_constant_sequence f ->
    eventually_periodic_sequence f.
Proof.
  intros A f [T [a Hconst]].
  exists T, 1%nat.
  split.
  - lia.
  - intros t Ht.
    assert (Ht1 : (T <= t + 1)%nat) by lia.
    rewrite (Hconst (t + 1)%nat Ht1).
    rewrite (Hconst t Ht).
    reflexivity.
Qed.

(*
  Width-w time-window extracted from a base sequence.
*)

Definition bounded_window_trace {A : Type} (f : nat -> A) (w t : nat) : list A :=
  map (fun k => f (t + k)%nat) (seq 0 w).

(*
  Deterministic image maps preserve eventual periodicity.
*)

Lemma eventually_periodic_mapped_trace :
  forall (A B : Type) (f : nat -> A) (g : A -> B),
    eventually_periodic_sequence f ->
    eventually_periodic_sequence (fun t => g (f t)).
Proof.
  intros A B f g [T [P [HP Hper]]].
  exists T, P.
  split.
  - exact HP.
  - intros t Ht.
    rewrite Hper by exact Ht.
    reflexivity.
Qed.

(*
  Bounded-window extraction preserves eventual periodicity.
*)

Lemma eventually_periodic_bounded_window_trace :
  forall (A : Type) (f : nat -> A) w,
    eventually_periodic_sequence f ->
    eventually_periodic_sequence (bounded_window_trace f w).
Proof.
  intros A f w [T [P [HP Hper]]].
  exists T, P.
  split.
  - exact HP.
  - intros t Ht.
    unfold bounded_window_trace.
    apply map_ext_in.
    intros k Hk.
    replace (t + P + k)%nat with ((t + k) + P)%nat by lia.
    apply Hper.
    lia.
Qed.

(*
  Bounded-window deterministic codings preserve eventual periodicity.
*)

Lemma eventually_periodic_bounded_window_image :
  forall (A B : Type) (f : nat -> A) (w : nat) (g : list A -> B),
    eventually_periodic_sequence f ->
    eventually_periodic_sequence (fun t => g (bounded_window_trace f w t)).
Proof.
  intros A B f w g Hperiodic.
  apply eventually_periodic_mapped_trace with
    (f := bounded_window_trace f w)
    (g := g).
  apply eventually_periodic_bounded_window_trace.
  exact Hperiodic.
Qed.

(*
  Tail anti-periodicity witness form.
*)

Definition tail_anti_periodic_sequence {A : Type} (f : nat -> A) : Prop :=
  forall T p,
    (0 < p)%nat ->
    exists n,
      (T <= n)%nat /\
      f n <> f (n + p)%nat.

(*
  Global anti-periodicity witness form.
*)

Definition anti_periodic_sequence {A : Type} (f : nat -> A) : Prop :=
  forall p,
    (0 < p)%nat ->
    exists n,
      f n <> f (n + p)%nat.

(*
  Tail anti-periodicity implies global anti-periodicity.
*)

Lemma tail_anti_periodic_implies_anti_periodic :
  forall (A : Type) (f : nat -> A),
    tail_anti_periodic_sequence f ->
    anti_periodic_sequence f.
Proof.
  intros A f Htail p Hp.
  destruct (Htail 0%nat p Hp) as [n [_ Hneq]].
  exists n.
  exact Hneq.
Qed.

(*
  Tail anti-periodicity excludes eventual periodicity.
*)

Lemma tail_anti_periodic_implies_not_eventually_periodic :
  forall (A : Type) (f : nat -> A),
    tail_anti_periodic_sequence f ->
    ~ eventually_periodic_sequence f.
Proof.
  intros A f Htail [T [P [HP Hper]]].
  destruct (Htail T P HP) as [n [Hn Hneq]].
  apply Hneq.
  symmetry.
  apply Hper.
  exact Hn.
Qed.

(*
  Classical converse: non-periodicity yields explicit tail mismatches.
*)

Lemma not_eventually_periodic_implies_tail_anti_periodic :
  forall (A : Type) (f : nat -> A),
    ~ eventually_periodic_sequence f ->
    tail_anti_periodic_sequence f.
Proof.
  intros A f Hnot T p Hp.
  apply NNPP.
  intro Hnone.
  apply Hnot.
  exists T, p.
  split.
  - exact Hp.
  - intros t Ht.
    destruct (classic (f (t + p)%nat = f t)) as [Heq | Hneq].
    + exact Heq.
    + exfalso.
      apply Hnone.
      exists t.
      split.
      * exact Ht.
      * intro Hsym.
        apply Hneq.
        symmetry.
        exact Hsym.
Qed.

(*
  Eventual periodicity for the compressed trace.

(*
  Eventual periodicity of the compressed trace.

Definition eventually_periodic_compressed_trace : Prop :=
  eventually_periodic_sequence compressed_trace.

(*

Definition eventually_periodic_packed_trace : Prop :=

(*

Definition eventually_periodic_centered_strip : Prop :=

(*
  Coordinate formula for z_segment lookup.
*)

Lemma nth_z_segment :
  forall start len i,
    (i < len)%nat ->
    nth i (z_segment start len) 0%Z = (start + Z.of_nat i)%Z.
Proof.
  intros start len.
  revert start.
  induction len as [|len IH]; intros start i Hi.
  - lia.
  - destruct i as [|i'].
    + simpl.
      lia.
    + simpl.
      assert (Hi' : (i' < len)%nat) by lia.
      rewrite (IH (start + 1)%Z i' Hi').
      replace (start + 1 + Z.of_nat i')%Z
        with (start + Z.of_nat (S i'))%Z by lia.
      reflexivity.
Qed.

(*
  In centered coordinates of width m, position m is coordinate 0.
*)

Lemma centered_coords_nth_center :
  forall m,
    nth m (centered_coords m) 0%Z = 0%Z.
Proof.
  intro m.
  unfold centered_coords.
  rewrite nth_z_segment.
  - replace (- Z.of_nat m + Z.of_nat m)%Z with 0%Z by lia.
    reflexivity.
  - lia.
Qed.

(*
  nth_error coordinate formula for z_segment lookup.
*)

Lemma nth_error_z_segment :
  forall start len i,
    (i < len)%nat ->
    nth_error (z_segment start len) i = Some (start + Z.of_nat i)%Z.
Proof.
  intros start len.
  revert start.
  induction len as [|len IH]; intros start i Hi.
  - lia.
  - destruct i as [|i'].
    + simpl.
      f_equal.
      lia.
    + simpl.
      assert (Hi' : (i' < len)%nat) by lia.
      rewrite (IH (start + 1)%Z i' Hi').
      replace (start + 1 + Z.of_nat i')%Z
        with (start + Z.of_nat (S i'))%Z by lia.
      reflexivity.
Qed.

(*
  nth_error form of the centered coordinate lookup at index m.
*)

Lemma nth_error_centered_coords_center :
  forall m,
    nth_error (centered_coords m) m = Some 0%Z.
Proof.
  intro m.
  unfold centered_coords.
  rewrite nth_error_z_segment.
  - replace (- Z.of_nat m + Z.of_nat m)%Z with 0%Z by lia.
    reflexivity.
  - lia.
Qed.

(*
  Option-valued center sample extracted from a centered strip row.
*)

Definition center_sample_from_strip (u : space_time) (m t : nat) : option cell :=
  nth_error (centered_strip u t m) m.

(*
  The compressed trace is determined from one centered strip row.

Theorem compressed_trace_depends_on_centered_strip :
  forall m t,
    center_sample_from_strip canonical_trajectory m t = Some (compressed_trace t).
Proof.
  intros m t.
  unfold center_sample_from_strip.
  unfold centered_strip, centered_strip_row.
  rewrite nth_error_map.
  rewrite nth_error_centered_coords_center.
  simpl.
  change (Some (center_bit t) = Some (compressed_trace t)).
  rewrite compressed_trace_of_center.
  reflexivity.
Qed.

(*
  If a centered strip is eventually periodic, then its extracted center
  sample sequence is eventually periodic.
*)

Lemma strip_periodicity_implies_center_sample_periodicity :
  forall m,
    eventually_periodic_strip canonical_trajectory m ->
    eventually_periodic_sequence (center_sample_from_strip canonical_trajectory m).
Proof.
  intros m [T [P [HP Hstrip]]].
  exists T, P.
  split.
  - exact HP.
  - intros t Ht.
    unfold center_sample_from_strip.
    specialize (Hstrip t Ht).
    now rewrite Hstrip.
Qed.

(*
  Periodicity transfer from centered strips to the compressed trace.
*)

Theorem strip_periodicity_implies_compressed_periodicity :
  forall m,
    eventually_periodic_strip canonical_trajectory m ->
    eventually_periodic_compressed_trace.
Proof.
  intros m Hstrip.
  destruct (strip_periodicity_implies_center_sample_periodicity m Hstrip)
    as [T [P [HP Hsample]]].
  exists T, P.
  split.
  - exact HP.
  - intros t Ht.
    specialize (Hsample t Ht).
    rewrite (compressed_trace_depends_on_centered_strip m (t + P)%nat) in Hsample.
    rewrite (compressed_trace_depends_on_centered_strip m t) in Hsample.
    inversion Hsample.
    reflexivity.
Qed.

Theorem strip_periodicity_implies_compression_periodicity :
  eventually_periodic_centered_strip ->
  eventually_periodic_packed_trace.
Proof.
  intros [m Hstrip].
  unfold eventually_periodic_packed_trace, packed_trace.
  pose proof (strip_periodicity_implies_compressed_periodicity m Hstrip)
    as Hcompressed.
  exact Hcompressed.
Qed.

(*
  The compressed-trace periodicity notion is equivalent to center periodicity.

Lemma eventually_periodic_center_implies_compressed_trace :
  eventually_periodic_center ->
  eventually_periodic_compressed_trace.
Proof.

  exists T, P.

  - intros t Ht.
    rewrite !compressed_trace_of_center.
    apply Hper.
    exact Ht.

  exists T, P.
  split.
  - exact HP.
  - intros t Ht.
    specialize (Hper t Ht).
    rewrite !compressed_trace_of_center in Hper.
    exact Hper.
Qed.

Lemma eventually_periodic_compressed_trace_iff_center :
  eventually_periodic_compressed_trace <-> eventually_periodic_center.
Proof.

  - apply eventually_periodic_compressed_trace_implies_center.
  - apply eventually_periodic_center_implies_compressed_trace.
Qed.

(*
  Compression definition: alias for the packed center-line trace.
*)

Definition compression_definition : nat -> bool :=
  packed_trace.

(*
  Boolean marker from a width-3 window: true iff pattern is [1;0;1].
*)

Definition window3_marker_of_list (xs : list bool) : bool :=
  match xs with
  | b0 :: b1 :: b2 :: nil => andb b0 (andb (negb b1) b2)
  | _ => false
  end.

(*
  Width-3 marker trace: applies the window3 marker to center bits.
*)

Definition window3_marker_trace (t : nat) : bool :=
  window3_marker_of_list (bounded_window_trace center_bit 3%nat t).

(*
  Prop-valued predicate: the center window at time t is [1;0;1].
*)

Definition window3_bad_local_configuration (t : nat) : Prop :=
  bounded_window_trace center_bit 3%nat t = [true; false; true].

(*
  Concrete expansion of the width-3 center window.
*)

Lemma bounded_window_trace_center_bit_3 :
  forall t,
    bounded_window_trace center_bit 3%nat t =
      [center_bit t; center_bit (S t); center_bit (t + 2)%nat].
Proof.
  intro t.
  unfold bounded_window_trace.
  simpl.
  replace (t + 0)%nat with t by lia.
  replace (t + 1)%nat with (S t) by lia.
  reflexivity.
Qed.

(*
  The two formulations of the center [1;0;1] pattern are equivalent.
*)

Lemma window3_bad_local_configuration_iff_center_window3_bad :
  forall t,
    window3_bad_local_configuration t <-> center_window3_bad t.
Proof.
  intro t.
  unfold window3_bad_local_configuration, center_window3_bad.
  rewrite bounded_window_trace_center_bit_3.
  split.
  - intro Hcfg.
    inversion Hcfg.
    repeat split; reflexivity.
  - intros [H0 [H1 H2]].
    rewrite H0, H1, H2.
    reflexivity.
Qed.

(*
  Characterization: window3_marker_of_list returns true iff input is [1;0;1].
*)

Lemma window3_marker_of_list_spec :
  forall xs,
    window3_marker_of_list xs = true <-> xs = [true; false; true].
Proof.
  intro xs.
  split.
  - destruct xs as [|b0 [|b1 [|b2 [|b3 xs]]]]; simpl; try discriminate.
    intro H.
    destruct b0, b1, b2; simpl in H; try discriminate; reflexivity.
  - intro H.
    subst xs.
    reflexivity.
Qed.

(*
  The boolean marker trace detects exactly the bad local configuration.
*)

Lemma window3_marker_trace_spec :
  forall t,
    window3_marker_trace t = true <-> window3_bad_local_configuration t.
Proof.
  intro t.
  unfold window3_marker_trace, window3_bad_local_configuration.
  apply window3_marker_of_list_spec.
Qed.

(*
  Tail anti-periodicity of the window3 marker trace.
*)

Definition window3_marker_direct_tail_killer : Prop :=
  tail_anti_periodic_sequence window3_marker_trace.

(*
  Local killer: for every candidate period p, there exists an
  arbitrarily late time where [1;0;1] occurs but not at offset p.
*)

Definition window3_marker_local_killer : Prop :=
  forall T p,
    (0 < p)%nat ->
    exists n,
      (T <= n)%nat /\
      window3_bad_local_configuration n /\
      ~ window3_bad_local_configuration (n + p)%nat.

Theorem window3_marker_local_killer_from_center_window3_bad_local_killer :
  (forall T p,
    (0 < p)%nat ->
    exists n,
      (T <= n)%nat /\
      center_window3_bad n /\
      ~ center_window3_bad (n + p)%nat) ->
  window3_marker_local_killer.
Proof.
  intros Hlocal T p Hp.
  destruct (Hlocal T p Hp) as [n [Hn [Hbad Hnotbad]]].
  exists n.
  split.
  - exact Hn.
  - split.
    + apply (proj2 (window3_bad_local_configuration_iff_center_window3_bad n)).
      exact Hbad.
    + intro Hcfg_np.
      apply Hnotbad.
      apply (proj1 (window3_bad_local_configuration_iff_center_window3_bad (n + p)%nat)).
      exact Hcfg_np.
Qed.

Theorem window3_marker_local_killer_from_r04_window3_package :
  center_window3_bad_occurs_arbitrarily_far ->
  center_window3_shift_impossible ->
  window3_marker_local_killer.
Proof.
  intros Hocc Hshift.
  apply window3_marker_local_killer_from_center_window3_bad_local_killer.
  apply (center_window3_bad_local_killer Hocc Hshift).
Qed.

Theorem window3_marker_period_killer_tail_of_local_killer :
  window3_marker_local_killer ->
  tail_anti_periodic_sequence window3_marker_trace.
Proof.
  intros Hlocal T p Hp.
  destruct (Hlocal T p Hp) as [n [Hn [Hbad Hnotbad]]].
  exists n.
  split.
  - exact Hn.
  - intro Heq.
    assert (Hmarker_n : window3_marker_trace n = true).
    { apply (proj2 (window3_marker_trace_spec n)).
      exact Hbad. }
    assert (Hmarker_np : window3_marker_trace (n + p)%nat = true).
    { rewrite <- Heq.
      exact Hmarker_n. }
    apply Hnotbad.
    apply (proj1 (window3_marker_trace_spec (n + p)%nat)).
    exact Hmarker_np.
Qed.

Section Periodicity_Utilities.

(*
  Eventual pure periodicity for a nat-indexed sequence.
*)

Definition eventually_periodic_sequence {A : Type} (f : nat -> A) : Prop :=
  exists T P : nat,
    (0 < P)%nat /\
    forall t,
      (T <= t)%nat ->
      f (t + P)%nat = f t.

(*
  Eventual constancy for a nat-indexed sequence.
*)

Definition eventually_constant_sequence {A : Type} (f : nat -> A) : Prop :=
  exists T : nat,
    exists a : A,
      forall t,
        (T <= t)%nat ->
        f t = a.

(*
  Deterministic image maps preserve eventual periodicity.
*)

Lemma eventually_periodic_mapped_trace :
  forall (A B : Type) (f : nat -> A) (g : A -> B),
    eventually_periodic_sequence f ->
    eventually_periodic_sequence (fun t => g (f t)).
Proof.
  intros A B f g [T [P [HP Hper]]].
  exists T, P.
  split.
  - exact HP.
  - intros t Ht.
    rewrite Hper by exact Ht.
    reflexivity.
Qed.

(*
  Alias for eventually_periodic_mapped_trace.
*)

Lemma eventually_periodic_map :
  forall (A B : Type) (f : nat -> A) (g : A -> B),
    eventually_periodic_sequence f ->
    eventually_periodic_sequence (fun t => g (f t)).
Proof.
  intros A B f g Hperiodic.
  apply eventually_periodic_mapped_trace.
  exact Hperiodic.
Qed.

(*
  Eventual constancy implies eventual periodicity (period 1).
*)

Lemma eventually_constant_implies_eventually_periodic :
  forall (A : Type) (f : nat -> A),
    eventually_constant_sequence f ->
    eventually_periodic_sequence f.
Proof.
  intros A f [T [a Hconst]].
  exists T, 1%nat.
  split.
  - lia.
  - intros t Ht.
    assert (Ht1 : (T <= t + 1)%nat) by lia.
    rewrite (Hconst (t + 1)%nat Ht1).
    rewrite (Hconst t Ht).
    reflexivity.
Qed.

(*
  Drop any finite prefix: eventual periodicity is preserved.
*)

Lemma eventually_periodic_sequence_drop_finite :
  forall (A : Type) (f : nat -> A) k,
    eventually_periodic_sequence f ->
    eventually_periodic_sequence (fun t => f (k + t)%nat).
Proof.
  intros A f k [T [P [HP Hper]]].
  exists T, P.
  split.
  - exact HP.
  - intros t Ht.
    replace (k + (t + P))%nat with ((k + t) + P)%nat by lia.
    apply Hper.
    lia.
Qed.

(*
  Alias for eventually_periodic_sequence_drop_finite.
*)

Lemma drop_prefix_preserves_eventual_periodicity :
  forall (A : Type) (f : nat -> A) k,
    eventually_periodic_sequence f ->
    eventually_periodic_sequence (fun t => f (k + t)%nat).
Proof.
  intros A f k Hperiodic.
  apply eventually_periodic_sequence_drop_finite.
  exact Hperiodic.
Qed.

(*
  Width-w time-window extracted from a base sequence.
*)

Definition bounded_window_trace {A : Type} (f : nat -> A) (w t : nat) : list A :=
  map (fun k => f (t + k)%nat) (seq 0 w).

(*
  Bounded-window extraction preserves eventual periodicity.
*)

Lemma eventually_periodic_bounded_window_trace :
  forall (A : Type) (f : nat -> A) w,
    eventually_periodic_sequence f ->
    eventually_periodic_sequence (bounded_window_trace f w).
Proof.
  intros A f w [T [P [HP Hper]]].
  exists T, P.
  split.
  - exact HP.
  - intros t Ht.
    unfold bounded_window_trace.
    apply map_ext_in.
    intros k Hk.
    replace (t + P + k)%nat with ((t + k) + P)%nat by lia.
    apply Hper.
    lia.
Qed.

(*
  Bounded-window deterministic codings preserve eventual periodicity.
*)

Lemma eventually_periodic_bounded_window_image :
  forall (A B : Type) (f : nat -> A) (w : nat) (g : list A -> B),
    eventually_periodic_sequence f ->
    eventually_periodic_sequence (fun t => g (bounded_window_trace f w t)).
Proof.
  intros A B f w g Hperiodic.
  apply eventually_periodic_mapped_trace with
    (f := bounded_window_trace f w)
    (g := g).
  apply eventually_periodic_bounded_window_trace.
  exact Hperiodic.
Qed.

(*
  Alias for eventually_periodic_bounded_window_image.
*)

Lemma bounded_window_map_preserves_periodicity :
  forall (A B : Type) (f : nat -> A) (w : nat) (g : list A -> B),
    eventually_periodic_sequence f ->
    eventually_periodic_sequence (fun t => g (bounded_window_trace f w t)).
Proof.
  intros A B f w g Hperiodic.
  apply eventually_periodic_bounded_window_image.
  exact Hperiodic.
Qed.

(*
  Tail anti-periodicity witness form.
*)

Definition tail_anti_periodic_sequence {A : Type} (f : nat -> A) : Prop :=
  forall T p,
    (0 < p)%nat ->
    exists n,
      (T <= n)%nat /\
      f n <> f (n + p)%nat.

(*
  Global anti-periodicity witness form.
*)

Definition anti_periodic_sequence {A : Type} (f : nat -> A) : Prop :=
  forall p,
    (0 < p)%nat ->
    exists n,
      f n <> f (n + p)%nat.

(*
  Tail anti-periodicity implies global anti-periodicity.
*)

Lemma tail_anti_periodic_implies_anti_periodic :
  forall (A : Type) (f : nat -> A),
    tail_anti_periodic_sequence f ->
    anti_periodic_sequence f.
Proof.
  intros A f Htail p Hp.
  destruct (Htail 0%nat p Hp) as [n [_ Hneq]].
  exists n.
  exact Hneq.
Qed.

(*
  A mismatch at/after a periodic tail start contradicts periodicity.
*)

Lemma periodic_tail_contradicts_mismatch :
  forall (A : Type) (f : nat -> A) T P,
    (0 < P)%nat ->
    (forall t, (T <= t)%nat -> f (t + P)%nat = f t) ->
    (exists n, (T <= n)%nat /\ f n <> f (n + P)%nat) ->
    False.
Proof.
  intros A f T P HP Hper [n [Hn Hneq]].
  apply Hneq.
  symmetry.
  apply Hper.
  exact Hn.
Qed.

(*
  Tail anti-periodicity excludes eventual periodicity.
*)

Lemma tail_anti_periodic_implies_not_eventually_periodic :
  forall (A : Type) (f : nat -> A),
    tail_anti_periodic_sequence f ->
    ~ eventually_periodic_sequence f.
Proof.
  intros A f Htail [T [P [HP Hper]]].
  destruct (Htail T P HP) as [n [Hn Hneq]].
  apply Hneq.
  symmetry.
  apply Hper.
  exact Hn.
Qed.

(*
  Classical converse: non-periodicity yields explicit tail mismatches.
*)

Lemma not_eventually_periodic_implies_tail_anti_periodic :
  forall (A : Type) (f : nat -> A),
    ~ eventually_periodic_sequence f ->
    tail_anti_periodic_sequence f.
Proof.
  intros A f Hnot T p Hp.
  apply NNPP.
  intro Hnone.
  apply Hnot.
  exists T, p.
  split.
  - exact Hp.
  - intros t Ht.
    destruct (classic (f (t + p)%nat = f t)) as [Heq | Hneq].
    + exact Heq.
    + exfalso.
      apply Hnone.
      exists t.
      split.
      * exact Ht.
      * intro Hsym.
        apply Hneq.
        symmetry.
        exact Hsym.
Qed.
